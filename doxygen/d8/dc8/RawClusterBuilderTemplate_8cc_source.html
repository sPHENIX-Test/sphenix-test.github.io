345 bool RawClusterBuilderTemplate::CorrectPhi(RawCluster *cluster, RawTowerContainer *towers, RawTowerGeomContainer *towergeom)
346 {
347  double sum = cluster->get_energy();
348  double phimin = 999.;
349  double phimax = -999.;
350  RawCluster::TowerConstRange begin_end = cluster->get_towers();
351  RawCluster::TowerConstIterator iter;
352  for (iter = begin_end.first; iter != begin_end.second; ++iter)
353  {
354  RawTower *tmpt = towers->getTower(iter->first);
355  double phi = towergeom->get_phicenter(tmpt->get_binphi());
356  if (phi > M_PI) phi = phi - 2. * M_PI; // correct the cluster phi for slat geometry which is 0-2pi (L. Xue)
357  if (phi < phimin)
358  {
359  phimin = phi;
360  }
361  if (phi > phimax)
362  {
363  phimax = phi;
364  }
365  }
366 
367  if ((phimax - phimin) < 3.) return false; // cluster is not at phi discontinuity
368 
369  float mean = 0.;
370  for (iter = begin_end.first; iter != begin_end.second; ++iter)
371  {
372  RawTower *tmpt = towers->getTower(iter->first);
373  double e = tmpt->get_energy();
374  double phi = towergeom->get_phicenter(tmpt->get_binphi());
375  if (phi > M_PI) phi = phi - 2. * M_PI; // correct the cluster phi for slat geometry which is 0-2pi (L. Xue)
376  if (phi < 0.)
377  {
378  phi = phi + 2. * M_PI; // shift phi range for correct mean calculation
379  }
380  mean += e * phi;
381  }
382  mean = mean / sum;
383  if (mean > M_PI)
384  {
385  mean = mean - 2. * M_PI; // shift back
386  }
387 
388  cluster->set_phi(mean);
389 
390  return true; // mean phi was corrected
391 }
