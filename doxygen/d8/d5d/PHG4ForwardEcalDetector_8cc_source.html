513 int PHG4ForwardEcalDetector::ParseParametersFromTable()
514 {
515  /* Open the datafile, if it won't open return an error */
516  ifstream istream_mapping;
517  istream_mapping.open(m_Params->get_string_param("mapping_file"));
518  if (!istream_mapping.is_open())
519  {
520  cout << "ERROR in PHG4ForwardEcalDetector: Failed to open mapping file " << m_Params->get_string_param("mapping_file") << endl;
521  gSystem->Exit(1);
522  }
523 
524  /* loop over lines in file */
525  string line_mapping;
526  while (getline(istream_mapping, line_mapping))
527  {
528  /* Skip lines starting with / including a '#' */
529  if (line_mapping.find("#") != string::npos)
530  {
531  if (Verbosity() > 0)
532  {
533  cout << "PHG4ForwardEcalDetector: SKIPPING line in mapping file: " << line_mapping << endl;
534  }
535  continue;
536  }
537 
538  istringstream iss(line_mapping);
539 
540  /* If line starts with keyword Tower, add to tower positions */
541  if (line_mapping.find("Tower ") != string::npos)
542  {
543  unsigned idx_j, idx_k, idx_l;
544  G4double pos_x, pos_y, pos_z;
545  G4double size_x, size_y, size_z;
546  G4double rot_x, rot_y, rot_z;
547  int type;
548  string dummys;
549 
550  /* read string- break if error */
551  if (!(iss >> dummys >> type >> idx_j >> idx_k >> idx_l >> pos_x >> pos_y >> pos_z >> size_x >> size_y >> size_z >> rot_x >> rot_y >> rot_z))
552  {
553  cout << "ERROR in PHG4ForwardEcalDetector: Failed to read line in mapping file " << m_Params->get_string_param("mapping_file") << endl;
554  gSystem->Exit(1);
555  }
556 
557  /* Construct unique name for tower */
558  /* Mapping file uses cm, this class uses mm for length */
559  ostringstream towername;
560  towername << m_TowerLogicNamePrefix << "_t_" << type << "_j_" << idx_j << "_k_" << idx_k;
561  /* Add Geant4 units */
562  pos_x = pos_x * cm;
563  pos_y = pos_y * cm;
564  pos_z = pos_z * cm;
565 
566  /* insert tower into tower map */
567  towerposition tower_new;
568  tower_new.x = pos_x;
569  tower_new.y = pos_y;
570  tower_new.z = pos_z;
571  tower_new.type = type;
572  m_TowerPositionMap.insert(make_pair(towername.str(), tower_new));
573  }
574  else
575  {
576  /* If this line is not a comment and not a tower, save parameter as string / value. */
577  string parname;
578  double parval;
579 
580  /* read string- break if error */
581  if (!(iss >> parname >> parval))
582  {
583  cerr << "ERROR in PHG4ForwardEcalDetector: Failed to read line in mapping file " << m_Params->get_string_param("mapping_file") << endl;
584  gSystem->Exit(1);
585  }
586 
587  m_GlobalParameterMap.insert(make_pair(parname, parval));
588  }
589  }
590  /* Update member variables for global parameters based on parsed parameter file */
591  std::map<string, double>::iterator parit;
592  ostringstream twr;
593  for (int i = 0; i < 7; i++)
594  {
595  twr.str("");
596  twr << "Gtower" << i << "_dx";
597  parit = m_GlobalParameterMap.find(twr.str());
598  m_TowerDx[i] = parit->second * cm;
599  twr.str("");
600  twr << "Gtower" << i << "_dy";
601  parit = m_GlobalParameterMap.find(twr.str());
602  m_TowerDy[i] = parit->second * cm;
603  twr.str("");
604  twr << "Gtower" << i << "_dz";
605  parit = m_GlobalParameterMap.find(twr.str());
606  m_TowerDz[i] = parit->second * cm;
607  }
608  ostringstream rad;
609  for (int i = 0; i < 2; i++)
610  {
611  int index = i + 1;
612  rad.str("");
613  rad << "Gr" << index << "_inner";
614  parit = m_GlobalParameterMap.find(rad.str());
615  if (parit != m_GlobalParameterMap.end())
616  {
617  m_RMin[i] = parit->second * cm;
618  }
619  rad.str("");
620  rad << "Gr" << index << "_outer";
621  parit = m_GlobalParameterMap.find(rad.str());
622  if (parit != m_GlobalParameterMap.end())
623  {
624  m_RMax[i] = parit->second * cm;
625  }
626  }
627  parit = m_GlobalParameterMap.find("Gdz");
628  if (parit != m_GlobalParameterMap.end())
629  {
630  m_dZ = parit->second * cm;
631  }
632  parit = m_GlobalParameterMap.find("Gx0");
633  if (parit != m_GlobalParameterMap.end())
634  {
635  m_PlaceX = parit->second * cm;
636  }
637  parit = m_GlobalParameterMap.find("Gy0");
638  if (parit != m_GlobalParameterMap.end())
639  {
640  m_PlaceY = parit->second * cm;
641  }
642  parit = m_GlobalParameterMap.find("Gz0");
643  if (parit != m_GlobalParameterMap.end())
644  {
645  m_PlaceZ = parit->second * cm;
646  }
647  parit = m_GlobalParameterMap.find("Grot_x");
648  if (parit != m_GlobalParameterMap.end())
649  {
650  m_XRot = parit->second;
651  }
652  parit = m_GlobalParameterMap.find("Grot_y");
653  if (parit != m_GlobalParameterMap.end())
654  {
655  m_YRot = parit->second;
656  }
657  parit = m_GlobalParameterMap.find("Grot_z");
658  if (parit != m_GlobalParameterMap.end())

660  m_ZRot = parit->second;
661  }
662  return 0;
