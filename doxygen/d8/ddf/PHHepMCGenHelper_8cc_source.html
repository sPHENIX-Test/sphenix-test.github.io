167 double PHHepMCGenHelper::smear(const double position,
168  const double width,
169  VTXFUNC dist) const
170 {
171  assert(width >= 0);
172 
173  double res = position;
174 
175  if (width == 0)
176  return res;
177 
178  if (dist == Uniform)
179  {
180  res = (position - width) + 2 * gsl_rng_uniform_pos(RandomGenerator) * width;
181  }
182  else if (dist == Gaus)
183  {
184  res = position + gsl_ran_gaussian(RandomGenerator, width);
185  }
186  else
187  {
188  cout << "PHHepMCGenHelper::smear - FATAL Error - unknown vertex function " << dist << endl;
189  exit(10);
190  }
191  return res;
192 }
