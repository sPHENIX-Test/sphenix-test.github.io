448  << "vy = " << vector_fiber.y() << "mm, " //
449  << "vz = " << vector_fiber.z() << "mm, " //
450  << endl;
451 
452  const G4double rotation_angle = G4Vector3D(0, 0, 1).angle(
453  vector_fiber);
454  const G4Vector3D rotation_axis =
455  rotation_angle == 0 ? G4Vector3D(1, 0, 0) : G4Vector3D(0, 0, 1).cross(vector_fiber);
456 
457  G4Transform3D fiber_place(
458  G4Translate3D(center_fiber.x(), center_fiber.y(),
459  center_fiber.z()) *
460  G4Rotate3D(rotation_angle, rotation_axis));
461 
462  stringstream name;
463  name << GetName() + string("_Tower") << g_tower.id << "_fiber"
464  << ss.str();
465 
466  const bool overlapcheck_fiber = OverlapCheck() and (get_geom_v3()->get_construction_verbose() >= 3);
467  G4PVPlacement* fiber_physi = new G4PVPlacement(fiber_place,
468  fiber_logic, G4String(name.str().c_str()), LV_tower, false,
469  fiber_ID, overlapcheck_fiber);
470  fiber_vol[fiber_physi] = fiber_ID;
471 
472  assert(gdml_config);
473  gdml_config->exclude_physical_vol(fiber_physi);
474 
475  ++fiber_cnt;
476  }
477  }
478 
479  if (get_geom_v3()->get_construction_verbose() >= 3)
480  cout << "PHG4FullProjSpacalDetector::Construct_Fibers::" << GetName()
481  << " - constructed tower ID " << g_tower.id << " with " << fiber_cnt
482  << " fibers" << endl;
483 
484  return fiber_cnt;

488 G4LogicalVolume*
489 PHG4FullProjSpacalDetector::Construct_Tower(
490  const PHG4FullProjSpacalDetector::SpacalGeom_t::geom_tower& g_tower)
491 {
492  std::stringstream sout;
493  sout << "_" << g_tower.id;
494  const G4String sTowerID(sout.str());
495 
496  //Processed PostionSeeds 1 from 1 1
497 
498  G4Trap* block_solid = new G4Trap(
499  /*const G4String& pName*/ G4String(GetName().c_str()) + sTowerID,
500  g_tower.pDz * cm, // G4double pDz,
501  g_tower.pTheta * rad, g_tower.pPhi * rad, // G4double pTheta, G4double pPhi,
502  g_tower.pDy1 * cm, g_tower.pDx1 * cm, g_tower.pDx2 * cm, // G4double pDy1, G4double pDx1, G4double pDx2,
503  g_tower.pAlp1 * rad, // G4double pAlp1,
504  g_tower.pDy2 * cm, g_tower.pDx3 * cm, g_tower.pDx4 * cm, // G4double pDy2, G4double pDx3, G4double pDx4,
505  g_tower.pAlp2 * rad // G4double pAlp2 //
506  );
507 
508  G4Material* cylinder_mat = G4Material::GetMaterial(
509  get_geom_v3()->get_absorber_mat());
510  assert(cylinder_mat);
511 
512  G4LogicalVolume* block_logic = new G4LogicalVolume(block_solid, cylinder_mat,
513  G4String(G4String(GetName()) + string("_Tower") + sTowerID), 0, 0,
514  nullptr);
515 
516  GetDisplayAction()->AddVolume(block_logic, "Block");
517 
518  // construct fibers
519 
520  int fiber_count = 0;
521 
522  if (get_geom_v3()->get_config() == SpacalGeom_t::kFullProjective_2DTaper)
523  {
524  fiber_count = Construct_Fibers(g_tower, block_logic);
525 
526  if (get_geom_v3()->get_construction_verbose() >= 2)
527  cout << "PHG4FullProjSpacalDetector::Construct_Tower::" << GetName()
528  << " - constructed tower ID " << g_tower.id << " with "
529  << fiber_count << " fibers using Construct_Fibers" << endl;
530  }
531  else if (get_geom_v3()->get_config() == SpacalGeom_t::kFullProjective_2DTaper_SameLengthFiberPerTower)
532  {
533  fiber_count = Construct_Fibers_SameLengthFiberPerTower(g_tower,
534  block_logic);
535 
536  if (get_geom_v3()->get_construction_verbose() >= 2)
537  cout << "PHG4FullProjSpacalDetector::Construct_Tower::" << GetName()
538  << " - constructed tower ID " << g_tower.id << " with "
539  << fiber_count
540  << " fibers using Construct_Fibers_SameLengthFiberPerTower" << endl;
541  }
542  else
543  {
544  G4ExceptionDescription message;
545  message << "can not recognize configuration type " << get_geom_v3()->get_config();
546 
547  G4Exception("PHG4FullProjSpacalDetector::Construct_Tower", "Wrong",
548  FatalException, message, "");
549  }
550 
551  return block_logic;
