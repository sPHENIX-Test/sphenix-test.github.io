68 G4TBMagneticFieldSetup::G4TBMagneticFieldSetup(PHField * phfield)
69  : verbosity(0),
70  fChordFinder(0),
71  fStepper(0),
72  fIntgrDriver(0)
73 {
74  assert(phfield);
75 
76  fEMfield = new PHG4MagneticField(phfield);
77  fFieldMessenger = new G4TBFieldMessenger(this) ;
78  fEquation = new G4Mag_UsualEqRhs(fEMfield);
79  fMinStep = 0.005*mm ; // minimal step of 10 microns
80  fStepperType = 4 ; // ClassicalRK4 -- the default stepper
81 
82  fFieldManager = GetGlobalFieldManager();
83  UpdateField();
84  double point[4] = {0,0,0,0};
85  fEMfield->GetFieldValue(&point[0],&magfield_at_000[0]);
86  for (size_t i=0; i<sizeof(magfield_at_000)/sizeof(double);i++)
87  {
88  magfield_at_000[i] = magfield_at_000[i]/tesla;
89  }
90  if (verbosity > 0)
91  {
92  cout << "field: x" << magfield_at_000[0]
93  << ", y: " << magfield_at_000[1]
94  << ", z: " << magfield_at_000[2]
95  << endl;
96  }
97 }

179 //
180 // Register this field to 'global' Field Manager and
181 // Create Stepper and Chord Finder with predefined type, minstep (resp.)
182 //
183 
184 void G4TBMagneticFieldSetup::UpdateField()
185 {
186  SetStepper();
187 
188  fFieldManager->SetDetectorField(fEMfield );
189 
190  delete fChordFinder;
191 
192  fIntgrDriver = new G4MagInt_Driver(fMinStep,
193  fStepper,
194  fStepper->GetNumberOfVariables() );
195 
196  fChordFinder = new G4ChordFinder(fIntgrDriver);
197 
198  fFieldManager->SetChordFinder( fChordFinder );
199 }

202 //
203 // Set stepper according to the stepper type
204 //
205 
206 void G4TBMagneticFieldSetup::SetStepper()
207 {
208  G4int nvar = 8;
209 
210  delete fStepper;
211 
212  std::stringstream message;
213 
214  switch ( fStepperType )
215  {
216  case 0:
217  fStepper = new G4ExplicitEuler( fEquation, nvar );
218  message << "Stepper in use: G4ExplicitEuler";
219  break;
220  case 1:
221  fStepper = new G4ImplicitEuler( fEquation, nvar );
222  message << "Stepper in use: G4ImplicitEuler";
223  break;
224  case 2:
225  fStepper = new G4SimpleRunge( fEquation, nvar );
226  message << "Stepper in use: G4SimpleRunge";
227  break;
228  case 3:
229  fStepper = new G4SimpleHeum( fEquation, nvar );
230  message << "Stepper in use: G4SimpleHeum";
231  break;
232  case 4:
233  fStepper = new G4ClassicalRK4( fEquation, nvar );
234  message << "Stepper in use: G4ClassicalRK4 (default)";
235  break;
236  case 5:
237  fStepper = new G4CashKarpRKF45( fEquation, nvar );
238  message << "Stepper in use: G4CashKarpRKF45";
239  break;
240  case 6:
241  fStepper = nullptr; // new G4RKG3_Stepper( fEquation, nvar );
242  message << "G4RKG3_Stepper is not currently working for Magnetic Field";
243  break;
244  case 7:
245  fStepper = nullptr; // new G4HelixExplicitEuler( fEquation );
246  message << "G4HelixExplicitEuler is not valid for Magnetic Field";
247  break;
248  case 8:
249  fStepper = nullptr; // new G4HelixImplicitEuler( fEquation );
250  message << "G4HelixImplicitEuler is not valid for Magnetic Field";
251  break;
252  case 9:
253  fStepper = nullptr; // new G4HelixSimpleRunge( fEquation );
254  message << "G4HelixSimpleRunge is not valid for Magnetic Field";
255  break;
256  default: fStepper = nullptr;
257  }
258 
259  if (verbosity > 0) {
260  G4cout << " ---------- G4TBMagneticFieldSetup::SetStepper() -----------" << G4endl;
261  G4cout << " " << message.str() << endl;
262  G4cout << " Minimum step size: " << fMinStep/mm << " mm" << G4endl;
263  G4cout << " -----------------------------------------------------------" << G4endl;
264  }
265 
266  if (!fStepper)
267  {
268  cout << "no stepper set, edxiting now" << endl;
269  exit(1);
270  }
271 
272  return;
273 }

288 //
289 // Set the value of the Global Field value to fieldVector
290 //
291 
292 void G4TBMagneticFieldSetup::SetFieldValue(const G4ThreeVector fieldVector)
293 {
294  // Find the Field Manager for the global field
295  G4FieldManager* fieldMgr= GetGlobalFieldManager();
296 
297  if(fieldVector != G4ThreeVector(0.,0.,0.))
298  {
299  if(fEMfield) delete fEMfield;
300  fEMfield = new G4UniformMagField(fieldVector);
301 
302  fEquation->SetFieldObj(fEMfield); // must now point to the new field
303 
304  // UpdateField();
305 
306  fieldMgr->SetDetectorField(fEMfield);
307  }
308  else
309  {
310  // If the new field's value is Zero, then it is best to
311  // insure that it is not used for propagation.
312  delete fEMfield;
313  fEMfield = nullptr;
314  fEquation->SetFieldObj(fEMfield); // As a double check ...
315  fieldMgr->SetDetectorField(fEMfield);
316  }
317 }
