156 void FastTrackingEval::fill_tree(PHCompositeNode *topNode) {
157 
158  // Make sure to reset all the TTree variables before trying to set them.
159  reset_variables();
160  //std::cout << "A1" << std::endl;
161  event = _event;
162 
163  if (!_truth_container) {
164  LogError("_truth_container not found!");
165  return;
166  }
167 
168  if (!_trackmap) {
169  LogError("_trackmap not found!");
170  return;
171  }
172 
173  PHG4TruthInfoContainer::ConstRange range =
174  _truth_container->GetPrimaryParticleRange();
175  //std::cout << "A2" << std::endl;
176  for (PHG4TruthInfoContainer::ConstIterator truth_itr = range.first;
177  truth_itr != range.second; ++truth_itr) {
178 
179  PHG4Particle* g4particle = truth_itr->second;
180  if(!g4particle) {
181  LogDebug("");
182  continue;
183  }
184  //std::cout << "B1" << std::endl;
185 
186  SvtxTrack_FastSim* track = NULL;
187 
188  //std::cout << "TRACKmap size " << _trackmap->size() << std::endl;
189  for (SvtxTrackMap::ConstIter track_itr = _trackmap->begin();
190  track_itr != _trackmap->end();
191  track_itr++) {
192  //std::cout << "TRACK * " << track_itr->first << std::endl;
193  SvtxTrack_FastSim* temp = dynamic_cast<SvtxTrack_FastSim*>(track_itr->second);
194  if(!temp) {
195  std::cout << "ERROR CASTING PARTICLE!" << std::endl;
196  continue;
197  }
198  //std::cout << " PARTICLE!" << std::endl;
199 
200  if ((temp->get_truth_track_id() - g4particle->get_track_id()) == 0) {
201  track = temp;
202  }
203  }
204 
205  //std::cout << "B2" << std::endl;
206  gtrackID = g4particle->get_track_id();
207  gflavor = g4particle->get_pid();
208 
209  gpx = g4particle->get_px();
210  gpy = g4particle->get_py();
211  gpz = g4particle->get_pz();
212  gpt = sqrt(gpx*gpx+gpy*gpy);
213  gp = sqrt(gpx*gpx+gpy*gpy+gpz*gpz);
214  gtheta = atan(gpt/gpz);
215  if(gtheta<0)

218  }

220  {
221  geta=-log(gtheta/2);
222  }
223  gphi = atan(gpy/gpx);
224 
225  if (track) {
226  //std::cout << "C1" << std::endl;

235  p = sqrt(px*px+py*py+pz*pz);
236  theta = atan(pt/pz);
237  if(theta<0)
238  {
239  eta=log(-theta/2);
240  }
241  else
242  {
243  eta=-log(theta/2);
244  }
245  phi = atan(py/px);
246  dca2d = track->get_dca2d();
247 
248  TVector3 truth_mom(gpx,gpy,gpz);
249  TVector3 reco_mom(px, py, pz);
250  //std::cout << "C2" << std::endl;
251 
252  _h2d_Delta_mom_vs_truth_mom->Fill(truth_mom.Mag(), (reco_mom.Mag()-truth_mom.Mag())/truth_mom.Mag());
253  _h2d_Delta_mom_vs_truth_eta->Fill(truth_mom.Eta(), (reco_mom.Mag()-truth_mom.Mag())/truth_mom.Mag());
254  }
255  //std::cout << "B3" << std::endl;
256

258 
259  }
