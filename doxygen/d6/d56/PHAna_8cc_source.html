81 void PHAna::fill_truth_ntuple(PHCompositeNode *topNode)
82 {

93  ntvars[0] = _event;

97  ntvars[3] = sqrt(Square(particle->get_px())+Square(particle->get_py()));
98  ntvars[4] = atan2(particle->get_py(), particle->get_px());
