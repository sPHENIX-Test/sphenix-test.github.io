247  PHG4HitContainer *g4hit = findNode::getClass<PHG4HitContainer>(topNode, hitnodename.c_str());
248  if (!g4hit)
249  {
250  cout << "Could not locate g4 hit node " << hitnodename << endl;
251  gSystem->Exit(1);
252  }
253 
254  PHG4HitContainer::ConstIterator hiter;
255  PHG4HitContainer::ConstRange hit_begin_end = g4hit->getHits();
256 
257  double ihit = 0;
258  for (hiter = hit_begin_end.first; hiter != hit_begin_end.second; ++hiter)
259  {
260  double t0 = fmax(hiter->second->get_t(0), hiter->second->get_t(1));
261  if (t0 > max_time)
262  {
263  continue;
264  }
265 
266  // for very high occupancy events, accessing the TrkrHitsets on the node tree for every drifted electron seems to be very slow
267  // Instead, use a temporary map to accumulate the charge from all drifted electrons, then copy to the node tree later
268 
269  double eion = hiter->second->get_eion();
270  unsigned int n_electrons = gsl_ran_poisson(RandomGenerator, eion * electrons_per_gev);
271  if (Verbosity() > 100)
272  cout << " new hit with t0, " << t0 << " g4hitid " << hiter->first
273  << " eion " << eion << " n_electrons " << n_electrons
274  << " entry z " << hiter->second->get_z(0) << " exit z " << hiter->second->get_z(1) << " avg z" << (hiter->second->get_z(0) + hiter->second->get_z(1)) / 2.0
275  << endl;
276 
277  if (n_electrons <= 0)
278  {
279  if (n_electrons < 0)
280  {
281  cout << "really bad number of electrons: " << n_electrons
282  << ", eion: " << eion
283  << endl;
284  }
285  continue;
286  }
287 
288  if (Verbosity() > 100)
289  {
290  cout << endl
291  << "electron drift: g4hit " << hiter->first << " created electrons: " << n_electrons
292  << " from " << eion * 1000000 << " keV" << endl;
293  cout << " entry x,y,z = " << hiter->second->get_x(0) << " " << hiter->second->get_y(0) << " " << hiter->second->get_z(0)
294  << " radius " << sqrt(pow(hiter->second->get_x(0), 2) + pow(hiter->second->get_y(0), 2)) << endl;
295  cout << " exit x,y,z = " << hiter->second->get_x(1) << " " << hiter->second->get_y(1) << " " << hiter->second->get_z(1)
296  << " radius " << sqrt(pow(hiter->second->get_x(1), 2) + pow(hiter->second->get_y(1), 2)) << endl;
297  }
298 
299  for (unsigned int i = 0; i < n_electrons; i++)
300  {
301  // We choose the electron starting position at random from a flat distribution along the path length
302  // the parameter t is the fraction of the distance along the path betwen entry and exit points, it has values between 0 and 1
303  double f = gsl_ran_flat(RandomGenerator, 0.0, 1.0);
304 
305  double x_start = hiter->second->get_x(0) + f * (hiter->second->get_x(1) - hiter->second->get_x(0));
306  double y_start = hiter->second->get_y(0) + f * (hiter->second->get_y(1) - hiter->second->get_y(0));
307  double z_start = hiter->second->get_z(0) + f * (hiter->second->get_z(1) - hiter->second->get_z(0));
308  double t_start = hiter->second->get_t(0) + f * (hiter->second->get_t(1) - hiter->second->get_t(0));
309 
310  double radstart = sqrt(x_start * x_start + y_start * y_start);
311  double r_sigma = diffusion_trans * sqrt(tpc_length / 2. - fabs(z_start));
312  double rantrans = gsl_ran_gaussian(RandomGenerator, r_sigma);
313  rantrans += gsl_ran_gaussian(RandomGenerator, added_smear_sigma_trans);
314 
315  double t_path = (tpc_length / 2. - fabs(z_start)) / drift_velocity;
316  double t_sigma = diffusion_long * sqrt(tpc_length / 2. - fabs(z_start)) / drift_velocity;
317  double rantime = gsl_ran_gaussian(RandomGenerator, t_sigma);
318  rantime += gsl_ran_gaussian(RandomGenerator, added_smear_sigma_long) / drift_velocity;
319  double t_final = t_start + t_path + rantime;
320 
321  double z_final;
322  if (z_start < 0)
323  z_final = -tpc_length / 2. + t_final * drift_velocity;
324  else
325  z_final = tpc_length / 2. - t_final * drift_velocity;
326 
327  if (t_final < min_time || t_final > max_time)
328  {
329  //cout << "skip this, t_final = " << t_final << " is out of range " << min_time << " to " << max_time << endl;
330  continue;
331  }
332  double ranphi = gsl_ran_flat(RandomGenerator, -M_PI, M_PI);
333  double x_final = x_start + rantrans * cos(ranphi);
334  double y_final = y_start + rantrans * sin(ranphi);
335  double rad_final = sqrt(x_final * x_final + y_final * y_final);
336  // remove electrons outside of our acceptance. Careful though, electrons from just inside 30 cm can contribute in the 1st active layer readout, so leave a little margin
337  if (rad_final < min_active_radius - 2.0 || rad_final > max_active_radius + 1.0)
338  {
339  continue;
340  }
341 
342  if (Verbosity() > 1000)
343  {
344  cout << "electron " << i << " g4hitid " << hiter->first << " f " << f << endl;
345  cout << "radstart " << radstart << " x_start: " << x_start
346  << ", y_start: " << y_start
347  << ",z_start: " << z_start
348  << " t_start " << t_start
349  << " t_path " << t_path
350  << " t_sigma " << t_sigma
351  << " rantime " << rantime
352  << endl;
353 
354  //if( sqrt(x_start*x_start+y_start*y_start) > 68.0 && sqrt(x_start*x_start+y_start*y_start) < 72.0)
355  cout << " rad_final " << rad_final << " x_final " << x_final << " y_final " << y_final
356  << " z_final " << z_final << " t_final " << t_final << " zdiff " << z_final - z_start << endl;
357  }
358 
359  if (Verbosity() > 0)
360  nt->Fill(ihit, t_start, t_final, t_sigma, rad_final, z_start, z_final);
361 
362  // this fills the cells and updates the hits in temp_hitsetcontainer for this drifted electron hitting the GEM stack
363  MapToPadPlane(x_final, y_final, z_final, hiter, ntpad, nthit);
364  } // end loop over electrons for this g4hit
365  ihit++;
366 
367  // transfer the hits from temp_hitsetcontainer to hitsetcontainer on the node tree
368  TrkrHitSetContainer::ConstRange temp_hitset_range = temp_hitsetcontainer->getHitSets(TrkrDefs::TrkrId::tpcId);
369  for (TrkrHitSetContainer::ConstIterator temp_hitset_iter = temp_hitset_range.first;
370  temp_hitset_iter != temp_hitset_range.second;
371  ++temp_hitset_iter)
372  {
373  // we have an itrator to one TrkrHitSet for the Tpc from the temp_hitsetcontainer
374  TrkrDefs::hitsetkey node_hitsetkey = temp_hitset_iter->first;
375  const unsigned int layer = TrkrDefs::getLayer(node_hitsetkey);
376  const int sector = TpcDefs::getSectorId(node_hitsetkey);
377  const int side = TpcDefs::getSide(node_hitsetkey);
378  if(Verbosity()>100)
379  cout << "PHG4TpcElectronDrift: temp_hitset with key: " << node_hitsetkey << " in layer " << layer << " with sector " << sector << " side " << side << endl;
380 
381  // find or add this hitset on the node tree
382  TrkrHitSetContainer::Iterator node_hitsetit = hitsetcontainer->findOrAddHitSet(node_hitsetkey);
383 
384  // get all of the hits from the temporary hitset
385  TrkrHitSet::ConstRange temp_hit_range = temp_hitset_iter->second->getHits();
386  for(TrkrHitSet::ConstIterator temp_hit_iter = temp_hit_range.first;
387  temp_hit_iter != temp_hit_range.second;
388  ++temp_hit_iter)

412  } // end loop over temp hits
413  } // end loop over temp hitsets
414

417 
418  } // end loop over g4hits

493 void PHG4TpcElectronDrift::SetDefaultParameters()
494 {
495  // Data on gasses @20 C and 760 Torr from the following source:
496  // http://www.slac.stanford.edu/pubs/icfa/summer98/paper3/paper3.pdf
497  // diffusion and drift velocity for 400kV for NeCF4 90/10 from calculations:
498  // https://www.phenix.bnl.gov/WWW/p/draft/prakhar/tpc/HTML_Gas_Linear/Ne_CF4_90_10.html
499  double Ne_dEdx = 1.56; // keV/cm
