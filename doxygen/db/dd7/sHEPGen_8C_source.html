52  _p4_electron_lab = new HLorentzVector( 0.,
53  0.,
54  _p_electron_lab,

58  0.,

61 
62  /* The current version of HEPGen supports only a "Fixed Target" mode, i.e.
63  the target (proton) is assumed at rest. Therefore, need to boost collision
64  from the laboratory frame to the proton-at-rest frame. */
65  _p4_hadron_lab_invert = new HLorentzVector( 0.,
66  0.,
67  -1 * _p_hadron_lab,
68  sqrt(_p_hadron_lab*_p_hadron_lab+mass_p*mass_p) );
69 
70  /* 4-Vectors of colliding electron and proton in proton-at-rest frame */
71  _p4_electron_prest = new HLorentzVector( *_p4_electron_lab );
72  _p4_electron_prest->boost(sqrt(_p4_hadron_lab_invert->getQuare()),*_p4_hadron_lab_invert);
73 
74  _p4_hadron_prest = new HLorentzVector( *_p4_hadron_lab );
75  _p4_hadron_prest->boost(sqrt(_p4_hadron_lab_invert->getQuare()),*_p4_hadron_lab_invert);
76

78  {

80  _p4_electron_lab->print();

91  _p4_electron_prest->setVector( flip_pz );
