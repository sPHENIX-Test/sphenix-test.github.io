89 {
90  /* Check if there is an unused event left in input file */
91  if (entry >= nEntries)
92  {
93  if (filename.size() > 0)
94  {
95  cout << "input file " << filename << " exhausted" << endl;
96  }
97  else
98  {
99  cout << "no file opened" << endl;
100  }
101  return Fun4AllReturnCodes::ABORTRUN;
102  }
103 
104  /* Get event record from input file */
105  Tin->GetEntry(entry);
106 
107  /* Create GenEvent */
108  HepMC::GenEvent* evt = new HepMC::GenEvent();
109 
110  /* define the units (Pythia uses GeV and mm) */
111  evt->use_units(HepMC::Units::GEV, HepMC::Units::MM);
112 
113  /* add global information to the event */
114  evt->set_event_number(entry);
115 
116  /* process ID from pythia */
117  evt->set_signal_process_id( GenEvent->GetProcess() );
118 
119  /* Set the PDF information */
120  HepMC::PdfInfo pdfinfo;
121  pdfinfo.set_x1( 1 );
122  pdfinfo.set_x2( GenEvent->GetX() );
123  pdfinfo.set_scalePDF( GenEvent->GetQ2() );
124  evt->set_pdf_info( pdfinfo );
125 
126  /* Loop over all particles for this event in input file and fill
127  * vector with HepMC particles */
128  vector< HepMC::GenParticle* > hepmc_particles;
129  vector< unsigned > origin_index;
130 
131  /* save pointers to beam particles */
132  HepMC::GenParticle *hepmc_beam1 = nullptr;
133  HepMC::GenParticle *hepmc_beam2 = nullptr;
134 
135  for (unsigned ii = 0; ii < GenEvent->GetNTracks(); ii++)
136  {
137  /* Get particle / track from event records.
138  * Class documentation for erhic::VirtualParticle at
139  * http://www.star.bnl.gov/~tpb/eic-smear/classerhic_1_1_virtual_particle.html */
140  erhic::ParticleMC * track_ii = GenEvent->GetTrack(ii);
141 
142  /* Create HepMC particle record */
143  HepMC::GenParticle *hepmcpart = new HepMC::GenParticle( HepMC::FourVector(track_ii->GetPx(),
144  track_ii->GetPy(),
145  track_ii->GetPz(),
146  track_ii->GetE()),
147  track_ii->Id());
148 
149  /* translate eic-smear status codes to HepMC status codes */
150  switch ( track_ii->GetStatus() )
151  {
152  case 1: hepmcpart->set_status( 1 ); break; // 'stable particle'
153 
154  case 21: hepmcpart->set_status( 3 ); break; // 'documentation line'
155 
156  default: hepmcpart->set_status( 0 ); break; // 'null entry'
157  }
158 
159  /* assume the first two particles are the beam particles (which getsHepMC status 4)*/
160  if ( ii < 2 )
161  hepmcpart->set_status( 4 );
162 
163  /* add particle information */
164  hepmcpart->setGeneratedMass( track_ii->GetM() );
165 
166  /* append particle to vector */
167  hepmc_particles.push_back(hepmcpart);
168  origin_index.push_back( track_ii->GetIndex() );
169 
170  /* if first particle, call this the first beam particle */
171  if ( ii == 0 )
172  hepmc_beam1 = hepmcpart;
173 
174  /* if second particle, call this the second beam particle */
175  if ( ii == 1 )
176  hepmc_beam2 = hepmcpart;
177  }
178 
179  /* Check if hepmc_particles and origin_index vectors are the same size */
180  if ( hepmc_particles.size() != origin_index.size() )
181  {
182  cout << "ReadEICFiles::process_event - Lengths of HepMC particles and Origin index vectors do not match!" << endl;
183 
184  delete evt;
185  return Fun4AllReturnCodes::ABORTRUN;
186  }
187 
188  /* add HepMC particles to Hep MC vertices; skip first two particles
189  * in loop, assuming that they are the beam particles */
190  vector< HepMC::GenVertex* > hepmc_vertices;
191 
192  for ( unsigned p = 2; p < hepmc_particles.size(); p++ )
193  {
194  HepMC::GenParticle *pp = hepmc_particles.at(p);
195 
196  /* continue if vertices for particle are already set */
197  if ( pp->production_vertex() && pp->end_vertex() )
198  continue;
199 
200  /* access mother particle vertex */
201  erhic::ParticleMC * track_pp = GenEvent->GetTrack(p);
202 
203  unsigned parent_index = track_pp->GetParentIndex();
204 
205  HepMC::GenParticle *pmother = nullptr;
206  for ( unsigned m = 0; m < hepmc_particles.size(); m++ )
207  {
208  if ( origin_index.at( m ) == parent_index )
209  {
210  pmother = hepmc_particles.at( m );
211  break;
212  }
213  }
214 
215  /* if mother does not exist: create new vertex and add this particle as outgoing to vertex */
216  if ( !pmother )
217  {
218  HepMC::GenVertex* hepmcvtx = new HepMC::GenVertex( HepMC::FourVector( track_pp->GetVertex()[0],
219  track_pp->GetVertex()[1],
220  track_pp->GetVertex()[2],
221  0 )
222  );
223  hepmc_vertices.push_back( hepmcvtx );
224  hepmcvtx->add_particle_out(pp);
225  continue;
226  }
227  /* if mother exists and has end vertex: add this particle as outgoing to the mother's end vertex */
228  else if ( pmother->end_vertex() )
229  {
230  pmother->end_vertex()->add_particle_out(pp);
231  }
232  /* if mother exists and has no end vertex: create new vertex */
233  else
234  {
235  HepMC::GenVertex* hepmcvtx = new HepMC::GenVertex( HepMC::FourVector( track_pp->GetVertex()[0],
236  track_pp->GetVertex()[1],
237  track_pp->GetVertex()[2],
238  0 )
239  );
240  hepmc_vertices.push_back( hepmcvtx );
241  hepmcvtx->add_particle_in(pmother);
242  pmother->end_vertex()->add_particle_out(pp);
243  }
244  }
245 
246  /* Add end vertex to beam particles if they don't have one yet */
247  for ( unsigned p = 0; p < 2; p++ )
248  {
249  HepMC::GenParticle *pp = hepmc_particles.at(p);
250 
251  if ( ! pp->end_vertex() )
252  {
253  /* create collision vertex */
254  HepMC::GenVertex* hepmcvtx = new HepMC::GenVertex( HepMC::FourVector( 0,
255  0,
256  0,
257  0 )
258  );
259  hepmc_vertices.push_back( hepmcvtx );
260  hepmcvtx->add_particle_in( pp );
261  }
262  }
263 
264  /* Check that all particles (except beam particles) have a production vertex */
265  for ( unsigned p = 2; p < hepmc_particles.size(); p++ )
266  {
267  if ( ! hepmc_particles.at(p)->production_vertex() )
268  {
269  cout << "ReadEICFiles::process_event - Missing production vertex for one or more non-beam particles!" << endl;
270  delete evt;
271  return Fun4AllReturnCodes::ABORTRUN;
272  }
273  }
274 
275  /* Add HepMC vertices to event */
276  for ( unsigned v = 0; v < hepmc_vertices.size(); v++ )
277  evt->add_vertex( hepmc_vertices.at(v) );
278 
279  /* set beam particles */
280  evt->set_beam_particles( hepmc_beam1, hepmc_beam2 );
281 
282  /* pass HepMC to PHNode*/
283  PHHepMCGenEvent * success = hepmc_helper . insert_event(evt);
284  if (!success) {
285  cout << "ReadEICFiles::process_event - Failed to add event to HepMC record!" << endl;
286  return Fun4AllReturnCodes::ABORTRUN;
287  }
288  /* Count up number of 'used' events from input file */
289  entry++;
290 
291  /* Done */
292  return 0;
293 }
