100 
101  if (whichactive > 0) // in sctintillator
102  {
103  /* Find indizes of sctintillator / tower containing this step */
104  FindTowerIndex(touch, idx_j, idx_k);
105  tower_id = touch->GetCopyNumber();
106  }
107  else
108  {
109  tower_id = touch->GetCopyNumber();
110  }
111 
112  /* Get energy deposited by this step */
113  G4double edep = aStep->GetTotalEnergyDeposit() / GeV;
114  G4double eion = (aStep->GetTotalEnergyDeposit() - aStep->GetNonIonizingEnergyDeposit()) / GeV;
115  G4double light_yield = 0;
116

118  const G4Track* aTrack = aStep->GetTrack();
119 
120  // if this block stops everything, just put all kinetic energy into edep
121  if (detector_->IsBlackHole())

124  G4Track* killtrack = const_cast<G4Track*>(aTrack);

129  if (detector_->IsActive())

131  int idx_l = -1;

133  bool geantino = false;

135  aTrack->GetParticleDefinition()->GetParticleName().find("geantino") != string::npos)

143 
144  switch (prePoint->GetStepStatus())
145  {
146  case fGeomBoundary:
147  case fUndefined:
148  if (!hit)
149  {
150  hit = new PHG4Hitv1();
151  }
152  hit->set_scint_id(tower_id);
153 
154  /* Set hit location (tower index) */
155  hit->set_index_j(idx_j);
156  hit->set_index_k(idx_k);
157  hit->set_index_l(idx_l);
158

160  hit->set_x(0, prePoint->GetPosition().x() / cm);
161  hit->set_y(0, prePoint->GetPosition().y() / cm);
162  hit->set_z(0, prePoint->GetPosition().z() / cm);
163 
164  /* Set hit time */
165  hit->set_t(0, prePoint->GetGlobalTime() / nanosecond);
166

169  /* set intial energy deposit */

172

240

243  hit->set_eion(hit->get_eion() + eion);

246  hit->set_light_yield(hit->get_light_yield() + light_yield);
247  }

294  // this means we have to delete it in the dtor. If this was
295  // the last hit we processed the memory is still allocated

297  }
298  }
299 
300  return true;
301  }
302  else
303  {
304  return false;
305  }

356 int PHG4ForwardHcalSteppingAction::ParseG4VolumeName(G4VPhysicalVolume* volume, int& j, int& k)
357 {
358  boost::char_separator<char> sep("_");
359  boost::tokenizer<boost::char_separator<char> > tok(volume->GetName(), sep);
360  boost::tokenizer<boost::char_separator<char> >::const_iterator tokeniter;
361  for (tokeniter = tok.begin(); tokeniter != tok.end(); ++tokeniter)
362  {
363  if (*tokeniter == "j")
364  {
365  ++tokeniter;
366  if (tokeniter == tok.end()) break;
367  j = boost::lexical_cast<int>(*tokeniter);
368  }
369  else if (*tokeniter == "k")
370  {
371  ++tokeniter;
372  if (tokeniter == tok.end()) break;
373  k = boost::lexical_cast<int>(*tokeniter);
374  }
375  }
376 
377  return 0;
