265 bool calc_dca3d_line(
266  double &dca_xy, double &dca_z, /*cm*/
267  const TVector3 &track_point, const TVector3 &track_direction, const TVector3 &vertex) {
268 
269  dca_xy = NAN;
270  dca_z = NAN;
271 
272  if(track_direction.Mag() == 0) return false;
273 
274  TVector3 PV = track_point - vertex;
275  if(PV.Mag() < 0.000001) {
276  dca_xy = 0;
277  dca_z = 0;
278  return true;
279  }
280 
281  TVector3 PVxMom = track_direction.Cross(PV);
282 
283  TVector3 PCA = track_direction;
284 
285  if(PVxMom.Mag() < 0.000001) {
286  std::cout << __FILE__ <<": " << __LINE__ << ": PVxMom.Mag2() < 0.000001" <<std::endl;
287  return false;
288  }
289 
290  PCA.Rotate(TMath::PiOver2(), PVxMom); // direction
291 
292  PCA.SetMag(1.);
293 
294  PCA.SetMag(PV.Dot(PCA));
295 
296  TVector3 r = track_direction.Cross(TVector3(0, 0, 1));
297  r.SetMag(1.);
298 
299  dca_xy = PCA.Dot(r);
300 
301  dca_z = PCA.Z();
302 
303  return true;
304 }

306 bool calc_dca_circle_line(
307  double &dca_xy, double &dca_z, /*cm*/
308  const TVector3 &track_point /*cm*/, const TVector3 &track_mom /*GeV/c*/, const TVector3 &vertex, /*cm*/
309  const double & q = 1./*e*/, const double & B = 1.4 /*Tesla*/) {
310 
311  double z0 = track_point.Z() - vertex.Z();
312  double r0 = (track_point-vertex).Perp();
313  double pz = track_mom.Z();
314  double pt = track_mom.Perp();
315 
316  if(pt == 0 || B ==0 || q==0) {
317  LogWarning("pt == 0 || B ==0 || q==0");
318  dca_z = NAN;
319  dca_xy = NAN;
320  return false;
321  }
322 
323  dca_z = z0 - pz/pt*r0;
324 
325  const double ten_over_c = 333.564095198; // GeV/c in cm*e*T
326  double R = pt / B / q * ten_over_c; // radius in cm
327 
328  // make 2d versions
329  TVector3 track_point_2d(track_point.X(),track_point.Y(),0);
330  TVector3 track_mom_2d(track_mom.X(), track_mom.Y(),0);
331  TVector3 vertex_2d(vertex.X(), vertex.Y(),0);
332 
333  // calc track circle center
334  TVector3 track_2d_circle_center = track_mom_2d; //copy mom_2d
335  track_2d_circle_center.RotateZ(TMath::PiOver2()); //rotate Pi/2
336  track_2d_circle_center.SetMag(R); // scale to R, R could be negtive
337  track_2d_circle_center += track_point_2d; //shift base to point_2d
338 
339  dca_xy = TMath::Abs(R) - (track_2d_circle_center-vertex_2d).Mag();
340 
341  return true;
