264 int PHG4ParticleGeneratorVectorMeson::process_event(PHCompositeNode *topNode)
265 {
266  if (!ineve) cout << " G4InEvent not found " << endl;
267 
268  // Generate a new set of vectors for the vector meson for each event
269  // These are the momentum and direction vectors for the pre-decay vector meson
270 
271  // taken randomly from a fitted pT distribution to Pythia Upsilons
272 
273  double pt = 0.0;
274  if (pt_max != pt_min)
275  {
276  pt = fpt->GetRandom();
277  }
278  else
279  {
280  pt = pt_min;
281  }
282  // taken randomly from a fitted rapidity distribution to Pythia Upsilons
283 
284  double y = 0.0;
285  if (y_max != y_min)
286  {
287  y = frap->GetRandom();
288  }
289  else
290  {
291  y = y_min;
292  }
293  // 0 and 2*M_PI identical, so use gsl_rng_uniform which excludes 1.0
294  double phi = (2.0 * M_PI) * gsl_rng_uniform(RandomGenerator);
295 
296  // The mass of the meson is taken from a Breit-Wigner lineshape
297 
298  double mnow = trand->BreitWigner(mass, m_Width);
299 
300  // Get the pseudorapidity, eta, from the rapidity, mass and pt
301 
302  double mt = sqrt(pow(mnow, 2) + pow(pt, 2));
303  double eta = asinh(sinh(y) * mt / pt);
304 
305  // Put it in a TLorentzVector
306 
307  TLorentzVector vm;
308  vm.SetPtEtaPhiM(pt, eta, phi, mnow);
309 
310  int vtxindex = -9;
311 
312  if (!ReuseExistingVertex(topNode))
313  {
314  // If not reusing existing vertex Randomly generate vertex position in z
315 
316  // mean width
317  vtx_x = smearvtx(_vertex_x, _vertex_width_x, _vertex_func_x);
318  vtx_y = smearvtx(_vertex_y, _vertex_width_y, _vertex_func_y);
319  vtx_z = smearvtx(_vertex_z, _vertex_width_z, _vertex_func_z);
320  }
321  vtx_x += _vertex_offset_x;
322  vtx_y += _vertex_offset_y;
323  vtx_z += _vertex_offset_z;
324 
325  for (std::map<unsigned int, std::string>::iterator it = decay1_names.begin(); it != decay1_names.end(); ++it)
326  {
327  unsigned int decay_id = it->first;
328  std::string decay1_name = it->second;
329  std::string decay2_name;
330  std::map<unsigned int, std::string>::iterator jt = decay2_names.find(decay_id);
331  std::map<unsigned int, double>::iterator xt = decay_vtx_offset_x.find(decay_id);
332  std::map<unsigned int, double>::iterator yt = decay_vtx_offset_y.find(decay_id);
333  std::map<unsigned int, double>::iterator zt = decay_vtx_offset_z.find(decay_id);
334 
335  if (jt != decay2_names.end() && xt != decay_vtx_offset_x.end() && yt != decay_vtx_offset_y.end() && zt != decay_vtx_offset_z.end())
336  {
337  decay2_name = jt->second;
338  set_decay_types(decay1_name, decay2_name);
339  set_existing_vertex_offset_vector(xt->second, yt->second, zt->second);
340  }
341  else
342  {
343  cout << PHWHERE << "Decay particles && vertex info can't be found !!" << endl;
344  exit(1);
345  }
346 
347  // 3D Randomized vertex
348  if ((_vertex_size_width > 0.0) || (_vertex_size_mean != 0.0))
349  {
350  _vertex_size_mean = sqrt(pow(vtx_x, 2) + pow(vtx_y, 2) + pow(vtx_z, 2));
351  double r = smearvtx(_vertex_size_mean, _vertex_size_width, _vertex_size_func_r);
352  double x1 = 0.0;
353  double y1 = 0.0;
354  double z1 = 0.0;
355  gsl_ran_dir_3d(RandomGenerator, &x1, &y1, &z1);
356  x1 *= r;
357  y1 *= r;
358  z1 *= r;
359  vtxindex = ineve->AddVtx(vtx_x + x1, vtx_y + y1, vtx_z + z1, t0);
360  }
361  else if (decay_id == 0)
362  {
363  vtxindex = ineve->AddVtx(vtx_x, vtx_y, vtx_z, t0);
364  }
365 
366  // Now decay it
367  // Get the decay energy and momentum in the frame of the vector meson - this correctly handles decay particles of any mass.
368 
369  double E1 = (pow(mnow, 2) - pow(m2, 2) + pow(m1, 2)) / (2.0 * mnow);
370  double p1 = sqrt((pow(mnow, 2) - pow(m1 + m2, 2)) * (pow(mnow, 2) - pow(m1 - m2, 2))) / (2.0 * mnow);
371 
372  // In the frame of the vector meson, get a random theta and phi angle for particle 1
373  // Assume angular distribution in the frame of the decaying meson that is uniform in phi and goes as sin(theta) in theta
374  // particle 2 has particle 1 momentum reflected through the origin
375 
376  double th1 = fsin->GetRandom();
377  // 0 and 2*M_PI identical, so use gsl_rng_uniform which excludes 1.0
378  double phi1 = 2.0 * M_PI * gsl_rng_uniform(RandomGenerator);
379 
380  // Put particle 1 into a TLorentzVector
381 
382  double px1 = p1 * sin(th1) * cos(phi1);
383  double py1 = p1 * sin(th1) * sin(phi1);
384  double pz1 = p1 * cos(th1);
385  TLorentzVector v1;
386  v1.SetPxPyPzE(px1, py1, pz1, E1);
387 
388  // now boost the decay product v1 into the lab using a vector consisting of the beta values of the vector meson
389  // where p/E is v/c if we use GeV/c for p and GeV for E
390 
391  double betax = vm.Px() / vm.E();
392  double betay = vm.Py() / vm.E();
393  double betaz = vm.Pz() / vm.E();
394  v1.Boost(betax, betay, betaz);
395 
396  // The second decay product's lab vector is the difference between the original meson and the boosted decay product 1
397 
398  TLorentzVector v2 = vm - v1;
399 
400  // Add the boosted decay particles to the particle list for the event
401 
402  AddParticle(decay1_name, v1.Px(), v1.Py(), v1.Pz());
403  AddParticle(decay2_name, v2.Px(), v2.Py(), v2.Pz());
404 
405  // Now output the list of boosted decay particles to the node tree
406 
407  vector<PHG4Particle *>::const_iterator iter;
408  for (iter = particlelist.begin(); iter != particlelist.end(); ++iter)
409  {
410  PHG4Particle *particle = new PHG4Particlev1(*iter);
411  SetParticleId(particle, ineve);
412  ineve->AddParticle(vtxindex, particle);
413  if (embedflag != 0)
414  {
415  ineve->AddEmbeddedParticle(particle, embedflag);
416  }
417  }
418  // List what has been put into ineve for this event
419 
420  if (Verbosity() > 0)
421  {
422  ineve->identify();
423 
424  // Print some check output
425  cout << endl
426  << "Output some sanity check info from PHG4ParticleGeneratorVectorMeson:" << endl;
427 
428  cout << " Vertex for this event (X,Y,Z) is (" << vtx_x << ", " << vtx_y << ", " << vtx_z << ")" << endl;
429  // Print the decay particle kinematics
430 
431  cout << " Decay particle 1:"
432  << " px " << v1.Px()
433  << " py " << v1.Py()
434  << " pz " << v1.Pz()
435  << " eta " << v1.PseudoRapidity()
436  << " phi " << v1.Phi()
437  << " theta " << v1.Theta()
438  << " pT " << v1.Pt()
439  << " mass " << v1.M()
440  << " E " << v1.E()
441  << endl;
442 
443  cout << " Decay particle 2:"
444  << " px " << v2.Px()
445  << " py " << v2.Py()
446  << " pz " << v2.Pz()
447  << " eta " << v2.PseudoRapidity()
448  << " phi " << v2.Phi()
449  << " theta " << v2.Theta()
450  << " pT " << v2.Pt()
451  << " mass " << v2.M()
452  << " E " << v2.E()
453  << endl;
454 
455  // Print the input vector meson kinematics
456  cout << " Vector meson input kinematics: mass " << vm.M()
457  << " px " << vm.Px()
458  << " py " << vm.Py()
459  << " pz " << vm.Pz()
460  << " eta " << vm.PseudoRapidity()
461  << " y " << vm.Rapidity()
462  << " pt " << vm.Pt()
463  << " E " << vm.E()
464  << endl;
465 
466  // Now, as a check, reconstruct the mass from the particle 1 and 2 kinematics
467 
468  TLorentzVector vreco = v1 + v2;
469 
470  cout << " Reco'd vector meson kinematics: mass " << vreco.M()
471  << " px " << vreco.Px()
472  << " py " << vreco.Py()
473  << " pz " << vreco.Pz()
474  << " eta " << vreco.PseudoRapidity()
475  << " y " << vreco.Rapidity()
476  << " pt " << vreco.Pt()
477  << " E " << vreco.E()
478  << endl;
479  }
480  } // decay particles
481 
482  // Reset particlelist for the next event
483  while (particlelist.begin() != particlelist.end())
484  {
485  delete particlelist.back();
486  particlelist.pop_back();
487  }
488 
489  return 0;
490 }

492 double
493 PHG4ParticleGeneratorVectorMeson::smearvtx(const double position, const double width, FUNCTION dist) const
494 {
495  double res = position;
496  if (dist == Uniform)
497  {
498  res = (position - m_Width) + 2 * gsl_rng_uniform_pos(RandomGenerator) * m_Width;
499  }
500  else if (dist == Gaus)
501  {
502  res = position + gsl_ran_gaussian(RandomGenerator, m_Width);
503  }
504  return res;
505 }
