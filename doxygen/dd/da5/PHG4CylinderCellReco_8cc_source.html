534  cell->add_light_yield(hiter->second->get_light_yield() * vdedx[i1]);
535  }
536  cell->add_shower_edep(hiter->second->get_shower_id(), hiter->second->get_edep() * vdedx[i1]);
537  // just a sanity check - we don't want to mess up by having Nan's or Infs in our energy deposition
538  if (!isfinite(hiter->second->get_edep() * vdedx[i1]))
539  {
540  cout << PHWHERE << " invalid energy dep " << hiter->second->get_edep()
541  << " or path length: " << vdedx[i1] << endl;
542  }
543  }
544  vphi.clear();
545  veta.clear();
546 
547  } // end loop over g4hits
548 
549  int numcells = 0;
550 
551  for (it = cellptmap.begin(); it != cellptmap.end(); ++it)
552  {
553  cells->AddCell(it->second);
554  numcells++;
555  if (Verbosity() > 1)
556  {
557  cout << "Adding cell in bin phi: " << PHG4CellDefs::EtaPhiBinning::get_phibin(it->second->get_cellid())
558  << " phi: " << geo->get_phicenter(PHG4CellDefs::EtaPhiBinning::get_phibin(it->second->get_cellid())) * 180. / M_PI
559  << ", eta bin: " << PHG4CellDefs::EtaPhiBinning::get_etabin(it->second->get_cellid())
560  << ", eta: " << geo->get_etacenter(PHG4CellDefs::EtaPhiBinning::get_etabin(it->second->get_cellid()))
561  << ", energy dep: " << it->second->get_edep()
562  << endl;
563  }
564  }
565 
566  if (Verbosity() > 0)
567  {
568  cout << Name() << ": found " << numcells << " eta/phi cells with energy deposition" << endl;
569  }
570  }
571 
572  else // ------ size binning ---------------------------------------------------------------
573  {
574  sizeiter = cell_size.find(*layer);
575  if (sizeiter == cell_size.end())
576  {
577  cout << "logical screwup!!! no sizes for layer " << *layer << endl;
578  exit(1);
579  }
580  double zstepsize = (sizeiter->second).second;
581  double phistepsize = phistep[*layer];
582 
583  for (hiter = hit_begin_end.first; hiter != hit_begin_end.second; hiter++)
584  {
585  sum_energy_before_cuts += hiter->second->get_edep();
586  // checking ADC timing integration window cut
587  if (hiter->second->get_t(0) > tmin_max[*layer].second) continue;
588  if (hiter->second->get_t(1) < tmin_max[*layer].first) continue;
589 
590  double xinout[2];
591  double yinout[2];
592  double px[2];
593  double py[2];
594  double phi[2];
595  double z[2];
596  double phibin[2];
597  double zbin[2];
598  if (Verbosity() > 0) cout << "--------- new hit in layer # " << *layer << endl;
599 
600  for (int i = 0; i < 2; i++)
601  {
602  xinout[i] = hiter->second->get_x(i);
603  yinout[i] = hiter->second->get_y(i);
604  px[i] = hiter->second->get_px(i);
605  py[i] = hiter->second->get_py(i);
606  phi[i] = atan2(hiter->second->get_y(i), hiter->second->get_x(i));
607  z[i] = hiter->second->get_z(i);
608  phibin[i] = geo->get_phibin(phi[i]);
609  zbin[i] = geo->get_zbin(hiter->second->get_z(i));
610 
611  if (Verbosity() > 0) cout << " " << i << " phibin: " << phibin[i] << ", phi: " << phi[i] << ", stepsize: " << phistepsize << endl;
612  if (Verbosity() > 0) cout << " " << i << " zbin: " << zbin[i] << ", z = " << hiter->second->get_z(i) << ", stepsize: " << zstepsize << " offset: " << zmin_max[*layer].first << endl;
613  }
614  // check bin range
615  if (phibin[0] < 0 || phibin[0] >= nphibins || phibin[1] < 0 || phibin[1] >= nphibins)
616  {
617  continue;
618  }
619  if (zbin[0] < 0 || zbin[0] >= nzbins || zbin[1] < 0 || zbin[1] >= nzbins)
620  {
621  continue;
622  }
623 
624  if (zbin[0] < 0)
625  {
626  hiter->second->identify();
627  continue;
628  }
629  sum_energy_g4hit += hiter->second->get_edep();
630 
631  int intphibin = phibin[0];
632  int intzbin = zbin[0];
633  int intphibinout = phibin[1];
634  int intzbinout = zbin[1];
635 
636  if (Verbosity() > 0)
637  {
638  cout << " phi bin range: " << intphibin << " to " << intphibinout << " phi: " << phi[0] << " to " << phi[1] << endl;
639  cout << " Z bin range: " << intzbin << " to " << intzbinout << " Z: " << z[0] << " to " << z[1] << endl;
640  cout << " phi difference: " << (phi[1] - phi[0]) * 1000. << " milliradians." << endl;
641  cout << " phi difference: " << 2.5 * (phi[1] - phi[0]) * 10000. << " microns." << endl;
642  cout << " path length = " << sqrt((xinout[1] - xinout[0]) * (xinout[1] - xinout[0]) + (yinout[1] - yinout[0]) * (yinout[1] - yinout[0])) << endl;
643  cout << " px = " << px[0] << " " << px[1] << endl;
644  cout << " py = " << py[0] << " " << py[1] << endl;
645  cout << " x = " << xinout[0] << " " << xinout[1] << endl;
646  cout << " y = " << yinout[0] << " " << yinout[1] << endl;
647  }
648 
649  // Determine all fired cells
650 
651  double ax = phi[0];
652  double ay = z[0];
653  double bx = phi[1];
654  double by = z[1];
655  if (intphibin > intphibinout)
656  {
657  int tmp = intphibin;
658  intphibin = intphibinout;
659  intphibinout = tmp;
660  }
661  if (intzbin > intzbinout)
662  {
663  int tmp = intzbin;
664  intzbin = intzbinout;
665  intzbinout = tmp;
666  }
667 
668  double trklen = sqrt((ax - bx) * (ax - bx) + (ay - by) * (ay - by));
669  // if entry and exit hit are the same (seems to happen rarely), trklen = 0
670  // which leads to a 0/0 and an NaN in edep later on
671  // this code does for particles in the same cell a trklen/trklen (vdedx[ii]/trklen)
672  // so setting this to any non zero number will do just fine
673  // I just pick -1 here to flag those strange hits in case I want to analyze them
674  // later on
675  if (trklen == 0)
676  {
677  trklen = -1.;
678  }
679  vector<int> vphi;
680  vector<int> vz;
681  vector<double> vdedx;
682 
683  if (intphibin == intphibinout && intzbin == intzbinout) // single cell fired
684  {
685  if (Verbosity() > 0)
686  {
687  cout << "SINGLE CELL FIRED: " << intphibin << " " << intzbin << endl;
688  }
689  vphi.push_back(intphibin);
690  vz.push_back(intzbin);
691  vdedx.push_back(trklen);
692  }
693  else
694  {
695  double phistep_half = geo->get_phistep() / 2.;
696  double zstep_half = geo->get_zstep() / 2.;
697  for (int ibp = intphibin; ibp <= intphibinout; ibp++)
698  {
699  double cx = geo->get_phicenter(ibp) - phistep_half;
700  double dx = geo->get_phicenter(ibp) + phistep_half;
701  for (int ibz = intzbin; ibz <= intzbinout; ibz++)
702  {
703  double cy = geo->get_zcenter(ibz) - zstep_half;
704  double dy = geo->get_zcenter(ibz) + zstep_half;
705  //cout << "##### line: " << ax << " " << ay << " " << bx << " " << by << endl;
706  //cout << "####### cell: " << cx << " " << cy << " " << dx << " " << dy << endl;
707  pair<bool, double> intersect = PHG4Utils::line_and_rectangle_intersect(ax, ay, bx, by, cx, cy, dx, dy);
708  if (intersect.first)
709  {
710  if (Verbosity() > 0) cout << "CELL FIRED: " << ibp << " " << ibz << " " << intersect.second << endl;
711  vphi.push_back(ibp);
712  vz.push_back(ibz);
713  vdedx.push_back(intersect.second);
714  }
715  }
716  }
717  }
718  if (Verbosity() > 0) cout << "NUMBER OF FIRED CELLS = " << vz.size() << endl;
719 
720  double tmpsum = 0.;
721  for (unsigned int ii = 0; ii < vz.size(); ii++)
722  {
723  tmpsum += vdedx[ii];
724  vdedx[ii] = vdedx[ii] / trklen;
725  if (Verbosity() > 0) cout << " CELL " << ii << " dE/dX = " << vdedx[ii] << endl;
726  }
727  if (Verbosity() > 0) cout << " TOTAL TRACK LENGTH = " << tmpsum << " " << trklen << endl;
728 
729  for (unsigned int i1 = 0; i1 < vphi.size(); i1++) // loop over all fired cells
730  {
731  int iphibin = vphi[i1];
732  int izbin = vz[i1];
733 
734  unsigned long long tmp = iphibin;
735  unsigned long long key = tmp << 32;
736  key += izbin;
737  if (Verbosity() > 1)
738  {
739  cout << " iphibin " << iphibin << " izbin " << izbin << " key 0x" << hex << key << dec << endl;
740  }
741  // check to see if there is already an entry for this cell
742  PHG4Cell *cell = nullptr;
743  it = cellptmap.find(key);
744 
745  if (it != cellptmap.end())
746  {
747  cell = it->second;
748  if (Verbosity() > 1)
749  {
750  cout << " add energy to existing cell for key = " << cellptmap.find(key)->first << endl;
751  }
752 
753  if (Verbosity() > 1 && hiter->second->has_property(PHG4Hit::prop_light_yield) && std::isnan(hiter->second->get_light_yield() * vdedx[i1]))
754  {
755  cout << " NAN lighy yield with vdedx[i1] = " << vdedx[i1]
756  << " and hiter->second->get_light_yield() = " << hiter->second->get_light_yield() << endl;
757  }
758  }
759  else
760  {
761  if (Verbosity() > 1)
762  {
763  cout << " did not find a previous entry for key = " << key << " create a new one" << endl;
764  }
765  PHG4CellDefs::keytype cellkey = PHG4CellDefs::SizeBinning::genkey(*layer, izbin, iphibin);
766  cell = new PHG4Cellv1(cellkey);
767  cellptmap[key] = cell;
768  }
769  if (!isfinite(hiter->second->get_edep() * vdedx[i1]))
770  {
771  cout << "hit 0x" << hex << hiter->first << dec << " not finite, edep: "
772  << hiter->second->get_edep() << " weight " << vdedx[i1] << endl;
773  }
774  cell->add_edep(hiter->first, hiter->second->get_edep() * vdedx[i1]);
775  cell->add_edep(hiter->second->get_edep() * vdedx[i1]); // add edep to cell
776  if (hiter->second->has_property(PHG4Hit::prop_light_yield))
777  {
778  cell->add_light_yield(hiter->second->get_light_yield() * vdedx[i1]);
779  if (Verbosity() > 1 && !std::isfinite(hiter->second->get_light_yield() * vdedx[i1]))
780  {
781  cout << " NAN lighy yield with vdedx[i1] = " << vdedx[i1]
782  << " and hiter->second->get_light_yield() = " << hiter->second->get_light_yield() << endl;
783  }
784  }
785  cell->add_shower_edep(hiter->second->get_shower_id(), hiter->second->get_edep() * vdedx[i1]);
786  }
787  vphi.clear();
788  vz.clear();
789 
790  } // end loop over hits
791 
792  int numcells = 0;
793 
794  for (it = cellptmap.begin(); it != cellptmap.end(); ++it)
795  {
796  cells->AddCell(it->second);
797  numcells++;
798  if (Verbosity() > 1)
799  {
800  cout << "Adding cell for key " << it->first << " in bin phi: " << PHG4CellDefs::SizeBinning::get_phibin(it->second->get_cellid())
801  << " phi: " << geo->get_phicenter(PHG4CellDefs::SizeBinning::get_phibin(it->second->get_cellid())) * 180. / M_PI
802  << ", z bin: " << PHG4CellDefs::SizeBinning::get_zbin(it->second->get_cellid())
803  << ", z: " << geo->get_zcenter(PHG4CellDefs::SizeBinning::get_zbin(it->second->get_cellid()))
804  << ", energy dep: " << it->second->get_edep()
805  << endl;
806  }
807  }
808 
809  if (Verbosity() > 0)
810  {
811  cout << "found " << numcells << " z/phi cells with energy deposition" << endl;
812  }
813  }
814 
815  //==========================================================
816  // now reset the cell map before moving on to the next layer
817  if (Verbosity() > 1)
818  {
819  cout << "cellptmap for layer " << *layer << " has final length " << cellptmap.size();
820  }
821  while (cellptmap.begin() != cellptmap.end())
822  {
823  // Assumes that memmory is freed by the cylinder cell container when it is destroyed
824  cellptmap.erase(cellptmap.begin());
825  }
826  if (Verbosity() > 1)
827  {
828  cout << " reset it to " << cellptmap.size() << endl;
829  }
830  }
831  if (chkenergyconservation)
832  {
833  CheckEnergy(topNode);
834  }
835 
836  return Fun4AllReturnCodes::EVENT_OK;
837 }

877 int PHG4CylinderCellReco::CheckEnergy(PHCompositeNode *topNode)
878 {
879  PHG4CellContainer *cells = findNode::getClass<PHG4CellContainer>(topNode, cellnodename);
880  double sum_energy_cells = 0.;
881  double sum_energy_stored_hits = 0.;
882  double sum_energy_stored_showers = 0.;
883  PHG4CellContainer::ConstRange cell_begin_end = cells->getCells();
884  PHG4CellContainer::ConstIterator citer;
885  for (citer = cell_begin_end.first; citer != cell_begin_end.second; ++citer)
886  {
887  sum_energy_cells += citer->second->get_edep();
888  PHG4Cell::EdepConstRange cellrange = citer->second->get_g4hits();
889  for (PHG4Cell::EdepConstIterator iter = cellrange.first; iter != cellrange.second; ++iter)
890  {
891  sum_energy_stored_hits += iter->second;
892  }
893  PHG4Cell::ShowerEdepConstRange shwrrange = citer->second->get_g4showers();
894  for (PHG4Cell::ShowerEdepConstIterator iter = shwrrange.first; iter != shwrrange.second; ++iter)
895  {
896  sum_energy_stored_showers += iter->second;
897  }
898  }
899  // the fractional eloss for particles traversing eta bins leads to minute rounding errors
900  if (sum_energy_stored_hits > 0)
901  {
902  if (fabs(sum_energy_cells - sum_energy_stored_hits) / sum_energy_cells > 1e-6)
903  {
904  cout << "energy mismatch between cell energy " << sum_energy_cells
905  << " and stored hit energies " << sum_energy_stored_hits
906  << endl;
907  }
908  }
909  if (sum_energy_stored_showers > 0)
910  {
911  if (fabs(sum_energy_cells - sum_energy_stored_showers) / sum_energy_cells > 1e-6)
912  {
913  cout << "energy mismatch between cell energy " << sum_energy_cells
914  << " and stored shower energies " << sum_energy_stored_showers
915  << endl;
916  }
917  }
918  if (fabs(sum_energy_cells - sum_energy_g4hit) / sum_energy_g4hit > 1e-6)
919  {
920  cout << "energy mismatch between cells: " << sum_energy_cells
921  << " and hits: " << sum_energy_g4hit
922  << " diff sum(cells) - sum(hits): " << sum_energy_cells - sum_energy_g4hit
923  << " cut val " << fabs(sum_energy_cells - sum_energy_g4hit) / sum_energy_g4hit
924  << endl;
925  cout << Name() << ":total energy for this event: " << sum_energy_g4hit << " GeV" << endl;
926  cout << Name() << ": sum cell energy: " << sum_energy_cells << " GeV" << endl;
927  cout << Name() << ": sum shower energy: " << sum_energy_stored_showers << " GeV" << endl;
928  cout << Name() << ": sum stored hit energy: " << sum_energy_stored_hits << " GeV" << endl;
929  cout << Name() << ": hit energy before cuts: " << sum_energy_before_cuts << " GeV" << endl;
930  return -1;
931  }
932  else
933  {
934  if (Verbosity() > 0)
935  {
936  cout << Name() << ":total energy for this event: " << sum_energy_g4hit << " GeV" << endl;
937  cout << Name() << ": sum cell energy: " << sum_energy_cells << " GeV" << endl;
938  cout << Name() << ": sum shower energy: " << sum_energy_stored_showers << " GeV" << endl;
939  cout << Name() << ": sum stored hit energy: " << sum_energy_stored_hits << " GeV" << endl;
940  cout << Name() << ": hit energy before cuts: " << sum_energy_before_cuts << " GeV" << endl;
941  }
942  }
943  return 0;
944 }
