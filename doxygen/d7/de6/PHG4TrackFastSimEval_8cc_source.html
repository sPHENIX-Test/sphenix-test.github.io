278 //----------------------------------------------------------------------------//
279 //-- fill_tree():
280 //-- Fill the trees with truth, track fit, and cluster information
281 //----------------------------------------------------------------------------//
282 void PHG4TrackFastSimEval::fill_vertex_tree(PHCompositeNode *topNode)
283 {
284  if (!_truth_container)
285  {
286  LogError("_truth_container not found!");
287  return;
288  }
289 
290  if (!_trackmap)
291  {
292  LogError("_trackmap not found!");
293  return;
294  }
295 
296  if (!_vertexmap)
297  {
298  return;
299  }
300 
301  for (SvtxVertexMap::Iter iter = _vertexmap->begin();
302  iter != _vertexmap->end();
303  ++iter)
304  {
305  SvtxVertex *vertex = iter->second;
306 
307  // Make sure to reset all the TTree variables before trying to set them.
308  reset_variables();
309  //std::cout << "A1" << std::endl;
310  event = _event;
311 
312  if (!vertex)
313  {
314  LogDebug("");
315  continue;
316  }
317 
318  //std::cout << "C1" << std::endl;
319  trackID = vertex->get_id();
320  ntracks = vertex->size_tracks();
321 
322  vx = vertex->get_x();
323  vy = vertex->get_y();
324  vz = vertex->get_z();
325  deltavx = sqrt(vertex->get_error(1, 1));
326  deltavy = sqrt(vertex->get_error(2, 2));
327  deltavz = sqrt(vertex->get_error(3, 3));
328 
329  // best matched vertex
330  PHG4VtxPoint *best_vtx = nullptr;
331  int best_n_match = -1;
332  map<PHG4VtxPoint *, int> vertex_match_map;
333  for (auto iter = vertex->begin_tracks(); iter != vertex->end_tracks(); ++iter)
334  {
335  const auto &trackID = *iter;
336  const auto trackIter = _trackmap->find(trackID);
337 
338  if (trackIter == _trackmap->end()) continue;
339 
340  SvtxTrack_FastSim *temp = dynamic_cast<SvtxTrack_FastSim *>(trackIter->second);
341 
342  if (!temp) continue;
343 
344  const auto g4trackID = temp->get_truth_track_id();
345  const PHG4Particle *g4particle = _truth_container->GetParticle(g4trackID);
346  assert(g4particle);
347  PHG4VtxPoint *vtx = _truth_container->GetVtx(g4particle->get_vtx_id());
348 
349  int n_match = ++vertex_match_map[vtx];
350 
351  if (n_match > best_n_match)
352  {
353  best_n_match = n_match;
354  best_vtx = vtx;
355  }
356  }
357  if (best_vtx)
358  {
359  gvx = best_vtx->get_x();
360  gvy = best_vtx->get_y();
361  gvz = best_vtx->get_z();
362  gvt = best_vtx->get_t();
363 
364  n_from_truth = best_n_match;
365  gtrackID = best_vtx->get_id();
366  }
367  _eval_tree_vertex->Fill();
368  }
369  //std::cout << "B3" << std::endl;
370 
371  return;
372 }
