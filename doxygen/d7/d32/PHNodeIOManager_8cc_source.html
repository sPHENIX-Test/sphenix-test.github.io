161 {
162  // The write function of the PHCompositeNode topNode will
163  // recursively call the write functions of its subnodes, thus

262 string
263 PHNodeIOManager::getBranchClassName(TBranch* branch)
264 {
265  // OK. Here all the game is to find out the name of the type
266  // contained in this branch. In ROOT pre-3.01/05 versions, all
267  // branches we used were of the same type = TBranchObject, so that
268  // was easy. Since version 3.01/05 ROOT introduced new branch style
269  // with some TBranchElement objects. So far so good. The problem is
270  // that I did not find a common way to grab the typename of the
271  // object contained in those branches, so I hereby use some durty if
272  // { } else if { } ...
273 
274 #if ROOT_VERSION_CODE >= ROOT_VERSION(3, 01, 5)
275  TBranchElement* be = dynamic_cast<TBranchElement*>(branch);
276 
277  if (be)
278  {
279  // TBranchElement has a nice GetClassName() method for us :
280  return be->GetClassName();
281  }
282 #endif
283 
284  TBranchObject* bo = dynamic_cast<TBranchObject*>(branch);
285  if (bo)
286  {
287  // For this one we need to go down a little before getting the
288  // name...
289  TLeafObject* leaf = static_cast<TLeafObject*>(branch->GetLeaf(branch->GetName()));
290  assert(leaf != 0);
291  return leaf->GetTypeName();
292  }
293  cout << PHWHERE << "Fatal error, dynamic cast of TBranchObject failed" << endl;
294  exit(1);
295 }

297 bool PHNodeIOManager::readEventFromFile(size_t requestedEvent)
298 {
299  // Se non c'e niente, non possiamo fare niente. Logisch, n'est ce
300  // pas?
301  if (!tree)
302  {
303  PHMessage("PHNodeIOManager::readEventFromFile", PHError,
304  "Tree not initialized.");
305  return false;
306  }
307 
308  int bytesRead;
309 
310  // Due to the current implementation of TBuffer>>(Long_t) we need
311  // to cd() in the current file before trying to fetch any event,
312  // otherwise mixing of reading 2.25/03 DST with writing some
313  // 3.01/05 trees will fail.
314  string currdir = gDirectory->GetPath();
315  TFile* file_ptr = gFile; // save current gFile
316  file->cd();
317 
318  if (requestedEvent)
319  {
320  if ((bytesRead = tree->GetEvent(requestedEvent)))
321  {
322  eventNumber = requestedEvent + 1;
323  }
324  }
325  else
326  {
327  bytesRead = tree->GetEvent(eventNumber++);
328  }
329 
330  gFile = file_ptr; // recover gFile
331  gROOT->cd(currdir.c_str());
332 
333  if (!bytesRead)
334  {
335  return false;
336  }
337  if (bytesRead == -1)
338  {
339  cout << PHWHERE << "Error: Input TTree corrupt, exiting now" << endl;
340  exit(1);
341  }
342  return true;

369 PHCompositeNode*
370 PHNodeIOManager::reconstructNodeTree(PHCompositeNode* topNode)
371 {
372  if (!file)
373  {
374  if (filename.empty())
375  {
376  cout << PHWHERE << "filename was never set" << endl;
377  }
378  else
379  {
380  cout << PHWHERE << "TFile " << filename << " NULL pointer" << endl;
381  }
382  return nullptr;
383  }
384 
385  tree = static_cast<TTree*>(file->Get(TreeName.c_str()));
386 
387  if (!tree)
388  {
389  cout << PHWHERE << "PHNodeIOManager::reconstructNodeTree : Root Tree "
390  << TreeName << " not found in file " << file->GetName() << endl;
391  return nullptr;
392  }
393 
394  // ROOT sucks, we need a unique name for the tree so we can open multiple
395  // files. So we take the memory location of the file pointer which
396  // should be unique within this process to create it
397  ostringstream nname;
398  nname << TreeName << file;
399 
400  tree->SetName(nname.str().c_str());
401 
402  // Select the branches according to objectToRead
403  map<string, bool>::const_iterator it;
404 
405  if (tree->GetNbranches() > 0)
406  {
407  for (it = objectToRead.begin(); it != objectToRead.end(); ++it)
408  {
409  tree->SetBranchStatus((it->first).c_str(),
410  static_cast<bool>(it->second));
411  }
412  }
413  // The file contains a TTree with a list of the TBranchObjects
414  // attached to it.
415  TObjArray* branchArray = tree->GetListOfBranches();
416 
417  // We need these in the loops down below...
418  size_t i, j;
419 
420  // If a topNode was provided, we can feed the iterator with it.
421  if (!topNode)
422  {
423  topNode = new PHCompositeNode("TOP"); // create topNode if we got a null pointer
424  }
425  PHNodeIterator nodeIter(topNode);
426 
427  // Loop over all branches in the tree. Each branch-name contains the
428  // full 'path' of composite-nodes in the original node tree. We
429  // split the name and reconstruct the tree.
430  string delimeters = phooldefs::branchpathdelim + phooldefs::legacypathdelims; // add old backslash for backward compat
431  for (i = 0; i < (size_t)(branchArray->GetEntriesFast()); i++)
432  {
433  string branchname = (*branchArray)[i]->GetName();
434  vector<string> splitvec;
435  boost::split(splitvec, branchname, boost::is_any_of(delimeters));
436  for (size_t ia = 1; ia < splitvec.size() - 1; ia++) // -1 so we skip the node name
437  {
438  if (!nodeIter.cd(splitvec[ia]))
439  {
440  nodeIter.addNode(new PHCompositeNode(splitvec[ia]));
441  nodeIter.cd(splitvec[ia]);
442  }
443  }
444  TBranch* thisBranch = (TBranch*) ((*branchArray)[i]);
445 
446  // Skip non-selected branches
447  if (thisBranch->TestBit(kDoNotProcess))
448  {
449  continue;
450  }
451 
452  string branchClassName = getBranchClassName(thisBranch);
453  string branchName = thisBranch->GetName();
454  fBranches[branchName] = thisBranch;
455 
456  assert(gROOT != 0);
457  TClass* thisClass = gROOT->GetClass(branchClassName.c_str());
458 
459  if (!thisClass)
460  {
461  cout << PHWHERE << endl;
462  cout << "Missing Class: " << branchClassName.c_str() << endl;
463  cout << "Did you forget to load the shared library which contains "
464  << branchClassName.c_str() << "?" << endl;
465  }
466  // it does not make sense to continue - the code coredumps
467  // later if a class is not loaded
468  assert(thisClass != 0);
469 
470  PHIODataNode<TObject>* newIODataNode =
471  static_cast<PHIODataNode<TObject>*>(nodeIter.findFirst("PHIODataNode", (*splitvec.rbegin()).c_str()));
472  if (!newIODataNode)
473  {
474  TObject* newTObject = static_cast<TObject*>(thisClass->New());
475  newIODataNode = new PHIODataNode<TObject>(newTObject, (*splitvec.rbegin()).c_str());
476  nodeIter.addNode(newIODataNode);
477  }
478  else
479  {
480  TObject* oldobject = newIODataNode->getData();
481  string oldclass = oldobject->ClassName();
482  if (oldclass != branchClassName)
483  {
484  cout << "You only have to worry if you get this message when reading parallel files"
485  << endl
486  << "if you get this when opening the 2nd, 3rd,... file" << endl
487  << "It looks like your objects are not of the same version in these files" << endl;
488  cout << PHWHERE << "Found object " << oldobject->ClassName()
489  << " in node tree but the file "
490  << filename << " contains a " << branchClassName
491  << " object. The object will be replaced without harming you" << endl;
492  cout << "CAVEAT: If you use local copies of pointers to data nodes" << endl
493  << "instead of searching the node tree you are in trouble now" << endl;
494  delete newIODataNode;
495  TObject* newTObject = static_cast<TObject*>(thisClass->New());
496  newIODataNode = new PHIODataNode<TObject>(newTObject, (*splitvec.rbegin()).c_str());
497  nodeIter.addNode(newIODataNode);
498  }
499  }
500 
501  if (thisClass->InheritsFrom("PHObject"))
502  {
503  newIODataNode->setObjectType("PHObject");
504  }
505  else
506  {
507  cout << PHWHERE << branchClassName.c_str()
508  << " inherits neither from PHTable nor from PHObject"
509  << " setting type to PHObject" << endl;
510  newIODataNode->setObjectType("PHObject");
511  }
512  thisBranch->SetAddress(&(newIODataNode->data));
513  for (j = 1; j < splitvec.size() - 1; j++)
514  {
515  nodeIter.cd("..");
516  }
517  }
518  return topNode;
519 }
