188  double ax,
189  double ay,
190  double bx,
191  double by,
192  double cx,
193  double cy,
194  double dx,
195  double dy)
196 {
197  // Find if a line segment limited by points A and B
198  // intersects line segment limited by points C and D.
199  // First check if an infinite line defined by A and B intersects
200  // segment (C,D). If h is from 0 to 1 line and line segment intersect
201  // Then check in intersection point is between C and D
202  double rx = NAN;
203  double ry = NAN;
204  double ex = bx - ax; // E=B-A
205  double ey = by - ay;
206  double fx = dx - cx; // F=D-C
207  double fy = dy - cy;
208  double px = -ey; // P
209  double py = ex;
210 
211  double bottom = fx * px + fy * py; // F*P
212  double gx = ax - cx; // A-C
213  double gy = ay - cy;
214  double top = gx * px + gy * py; // G*P
215 
216  double h = 99999.;
217  if (bottom != 0.)
218  {
219  h = top / bottom;
220  }
221 
222  //intersection point R = C + F*h
223  if (h > 0. && h < 1.)
224  {
225  rx = cx + fx * h;
226  ry = cy + fy * h;
227  //cout << " line/segment intersection coordinates: " << *rx << " " << *ry << endl;
228  if ((rx > ax && rx > bx) || (rx < ax && rx < bx) || (ry < ay && ry < by) || (ry > ay && ry > by))
229  {
230  //cout << " NO segment/segment intersection!" << endl;
231  return make_pair(false, make_pair(NAN, NAN));
232  }
233  else
234  {
235  //cout << " segment/segment intersection!" << endl;
236  return make_pair(true, make_pair(rx, ry));
237  }
238  }
239

241 }

335 double PHG4Utils::sA(double r, double x, double y)
336 {
337  // Uses analytic formula for the integral of a circle between limits set by the corner of a rectangle
338  // It is called repeatedly to find the overlap area between the circle and rectangle
339  // I found this code implementing the integral on a web forum called "ars technica",
340  // https://arstechnica.com/civis/viewtopic.php?t=306492
341  // posted by "memp"
342 
343  double a;
344 
345  if (x < 0)
346  {
347  return -sA(r, -x, y);
348  }
349 
350  if (y < 0)
351  {
352  return -sA(r, x, -y);
353  }
354 
355  if (x > r)
356  {
357  x = r;
358  }
359 
360  if (y > r)
361  {
362  y = r;
363  }
364 
365  if (x * x + y * y > r * r)
366  {
367  a = r * r * asin(x / r) + x * sqrt(r * r - x * x) + r * r * asin(y / r) + y * sqrt(r * r - y * y) - r * r * M_PI_2;
368 
369  a *= 0.5;
370  }
371  else
372  {
373  a = x * y;
374  }
375 
376  return a;
377 }
