87 bool twrs::is_adjacent(twrs &tower)
88 {
89  if (bineta - 1 <= tower.get_bineta() && tower.get_bineta() <= bineta + 1)
90  {
91  if (binphi - 1 <= tower.get_binphi() && tower.get_binphi() <= binphi + 1)
92  {
93  return true;
94  }
95  // cluster through the phi-wraparound

98  {
99  return true;
100  }
101  }
102 
103  return false;
104 }
