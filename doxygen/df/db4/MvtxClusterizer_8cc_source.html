228  hitr != hitrangei.second;
229  ++hitr)
230  {
231  hitvec.push_back(make_pair(hitr->first, hitr->second));
232  }
233  if (Verbosity() > 2)
234  cout << "hitvec.size(): " << hitvec.size() << endl;

236  // do the clustering
237  typedef adjacency_list<vecS, vecS, undirectedS> Graph;
238  Graph G;
239 
240  // loop over hits in this chip
241  for (unsigned int i = 0; i < hitvec.size(); i++)
242  {
243  for (unsigned int j = 0; j < hitvec.size(); j++)
244  {
245  if (are_adjacent(hitvec[i], hitvec[j]))
246  add_edge(i, j, G);
247  }
248  }
249 
250  // Find the connections between the vertices of the graph (vertices are the rawhits,
251  // connections are made when they are adjacent to one another)
252  vector<int> component(num_vertices(G));
253 
254  // this is the actual clustering, performed by boost
255  connected_components(G, &component[0]);
256 
257  // Loop over the components(hits) compiling a list of the
258  // unique connected groups (ie. clusters).
259  set<int> cluster_ids; // unique components
260  //multimap<int, pixel> clusters;
261  multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*> > clusters;
262  for (unsigned int i = 0; i < component.size(); i++)
263  {
264  cluster_ids.insert(component[i]);
265  clusters.insert(make_pair(component[i], hitvec[i]));
266  }
267 
268  // loop over the componenets and make clusters
269  for (set<int>::iterator clusiter = cluster_ids.begin(); clusiter != cluster_ids.end(); ++clusiter)
270  {
271  int clusid = *clusiter;
272  pair<multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*>>::iterator,
273  multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*>>::iterator> clusrange = clusters.equal_range(clusid);
274  multimap<int, std::pair<TrkrDefs::hitkey, TrkrHit*>>::iterator mapiter = clusrange.first;
275 
276  if (Verbosity() > 2)
277  cout << "Filling cluster id " << clusid << endl;
278 
279  // make the cluster directly in the node tree
280  TrkrDefs::cluskey ckey = MvtxDefs::genClusKey(hitset->getHitSetKey(), clusid);
281  TrkrClusterv1 *clus = static_cast<TrkrClusterv1 *>((m_clusterlist->findOrAddCluster(ckey))->second);
282 
283  // determine the size of the cluster in phi and z
284  set<int> phibins;
285  set<int> zbins;
286 
287  // determine the cluster position...
288  double xsum = 0.0;
289  double ysum = 0.0;
290  double zsum = 0.0;
291  unsigned nhits = 0;
292 
293  double clusx = NAN;
294  double clusy = NAN;
295  double clusz = NAN;
296 
297  // we need the geometry object for this layer to get the global positions
298  int layer = TrkrDefs::getLayer(ckey);
299  CylinderGeom_Mvtx *layergeom = dynamic_cast<CylinderGeom_Mvtx *>(geom_container->GetLayerGeom(layer));
300  if (!layergeom)
301  exit(1);
302 
303  int chip = MvtxDefs::getChipId(ckey);
304  int stave = MvtxDefs::getStaveId(ckey);
305 
306  for (mapiter = clusrange.first; mapiter != clusrange.second; ++mapiter)
307  {
308  // size
309  int col = MvtxDefs::getCol( (mapiter->second).first);
310  int row = MvtxDefs::getRow( (mapiter->second).first);
311  zbins.insert(col);
312  phibins.insert(row);
313 
314  int pixnum = layergeom->get_pixel_number_from_xbin_zbin(row, col);
315  //cout << " new mvtx clusterizer: cluster key " << ckey << " layer " << layer << " chip " << chip << " stave " << stave
316  // << " row " << row << " col " << col << " pixnum " << pixnum << endl;;
317 
318  TVector3 local_coords = layergeom->get_local_coords_from_pixel(pixnum);
319  TVector3 world_coords = layergeom->get_world_from_local_coords(stave, 0, 0, chip, local_coords);
320  //cout << " new: world coords: X " << world_coords.X() << " Y " << world_coords.Y() << " Z " << world_coords.Z() << endl;
321 
322  // find the center of the pixel in local coords
323  xsum += world_coords.X();
324  ysum += world_coords.Y();
325  zsum += world_coords.Z();
326 
327  // add the association between this cluster key and this hitkey to the table
328  m_clusterhitassoc->addAssoc(ckey, mapiter->second.first);
329 
330  ++nhits;
331  } //mapiter
332 
333 
334  // This is the global position
335  clusx = xsum / nhits;
336  clusy = ysum / nhits;
337  clusz = zsum / nhits;
338  //cout << "new mvtx clusterizer: clusx " << clusx << " clusy " << clusy << " clusz " << clusz << endl;
339  clus->setAdc(nhits);
340 
341  clus->setPosition(0, clusx);
342  clus->setPosition(1, clusy);
343  clus->setPosition(2, clusz);
344  clus->setGlobal();
345 
346  double thickness = layergeom->get_pixel_thickness();
347  double pitch = layergeom->get_pixel_x();
348  double length = layergeom->get_pixel_z();
349  double phisize = phibins.size() * pitch;
350  double zsize = zbins.size() * length;
351 
352  double ladder_location[3] = {0.0, 0.0, 0.0};
353  // returns the center of the sensor in world coordinates - used to get the ladder phi location
354  layergeom->find_sensor_center(stave, 0, 0, chip, ladder_location);
355  double ladderphi = atan2(ladder_location[1], ladder_location[0]);
356  ladderphi += layergeom->get_stave_phi_tilt();
357 
358  // tilt refers to a rotation around the radial vector from the origin, and this is zero for the MVTX ladders
359  float tilt = 0.0;
360 
361  double invsqrt12 = 1.0 / sqrt(12.0);
362 
363  TMatrixF DIM(3, 3);
364  DIM[0][0] = pow(0.5 * thickness, 2);
365  DIM[0][1] = 0.0;
366  DIM[0][2] = 0.0;
367  DIM[1][0] = 0.0;
368  DIM[1][1] = pow(0.5 * phisize, 2);
369  DIM[1][2] = 0.0;
370  DIM[2][0] = 0.0;
371  DIM[2][1] = 0.0;
372  DIM[2][2] = pow(0.5 * zsize, 2);
373 
374  TMatrixF ERR(3, 3);
375  ERR[0][0] = pow(0.5 * thickness * invsqrt12, 2);
376  ERR[0][1] = 0.0;
377  ERR[0][2] = 0.0;
378  ERR[1][0] = 0.0;
379  ERR[1][1] = pow(0.5 * phisize * invsqrt12, 2);
380  ERR[1][2] = 0.0;
381  ERR[2][0] = 0.0;
382  ERR[2][1] = 0.0;
383  ERR[2][2] = pow(0.5 * zsize * invsqrt12, 2);
384 
385  TMatrixF ROT(3, 3);
386  ROT[0][0] = cos(ladderphi);
387  ROT[0][1] = -1.0 * sin(ladderphi);
388  ROT[0][2] = 0.0;
389  ROT[1][0] = sin(ladderphi);
390  ROT[1][1] = cos(ladderphi);
391  ROT[1][2] = 0.0;
392  ROT[2][0] = 0.0;
393  ROT[2][1] = 0.0;
394  ROT[2][2] = 1.0;
395 
396  TMatrixF TILT(3, 3);
397  TILT[0][0] = 1.0;
398  TILT[0][1] = 0.0;
399  TILT[0][2] = 0.0;
400  TILT[1][0] = 0.0;
401  TILT[1][1] = cos(tilt);
402  TILT[1][2] = -1.0 * sin(tilt);
403  TILT[2][0] = 0.0;
404  TILT[2][1] = sin(tilt);
405  TILT[2][2] = cos(tilt);
406 
407  TMatrixF R(3, 3);
408  R = ROT * TILT;
409 
410  TMatrixF R_T(3, 3);
411  R_T.Transpose(R);
412 
413  TMatrixF COVAR_DIM(3, 3);
414  COVAR_DIM = R * DIM * R_T;
415

417  clus->setSize(0, 1, COVAR_DIM[0][1]);
