224  int np = m_himain1.np();
225  int nt = m_himain1.nt();
226  int n0 = m_himain1.n0();
227  int n01 = m_himain1.n01();
228  int n10 = m_himain1.n10();
229  int n11 = m_himain1.n11();
230 
231  // The Hijing documentation is a little unclear about this, but the
232  // four values, n0, n01, n10 and n11, count collisions between
233  // nucleons that have already (or have not already had) collisions
234  // with other nucleons. N_coll, the way we use it, is the sum of
235  // all these values.
236  int ncoll = n0 + n01 + n10 + n11;
237 
238  int natt = m_himain1.natt();
239  float b = m_hiparnt.hint1(19);
240  float bphi = m_hiparnt.hint1(20);
241 
242  // Determine the participant eccentricity.
243  std::vector<float> x;
244  std::vector<float> y;
245  float tx, ty;
246  float bbx = b * cos(bphi);
247  float bby = b * sin(bphi);
248  // std::cout << "x,y,c" << std::endl;
249  for (int i = 0; i < m_hiparnt.ihnt2(1); i++)
250  {
251  if (m_histrng.nfp(i,5) > 2)
252  {
253  tx = m_hijcrdn.yp(1,i)+bbx;
254  ty = m_hijcrdn.yp(2,i)+bby;
255  x.push_back(tx);
256  y.push_back(ty);
257 
258  // std::cout << tx << "," << ty << "," << "0" << std::endl;
259  }
260  }
261  for (int i = 0; i < m_hiparnt.ihnt2(3); i++)
262  {
263  if (m_histrng.nft(i,5) > 2)
264  {
265  tx = m_hijcrdn.yt(1,i);
266  ty = m_hijcrdn.yt(2,i);
267  x.push_back(tx);
268  y.push_back(ty);
269 
270  // std::cout << tx << "," << ty << "," << "1" << std::endl;
271  }
272  }
273 
274  float e_part = 0.0;
275  if (x.size() != 0)
276  {
277  float N = x.size();
278  float xa = std::accumulate(x.begin(), x.end(), 0) / N;
279  float ya = std::accumulate(y.begin(), y.end(), 0) / N;
280 
281  float sxx = 0;
282  float syy = 0;
283  float sxy = 0;
284  for (int i = 0; i < N; i++)
285  {
286  sxx += (x[i] - xa) * (x[i] - xa);
287  syy += (y[i] - ya) * (y[i] - ya);
288  sxy += (x[i] - xa) * (y[i] - ya);
289  }
290  sxx /= N;
291  syy /= N;
292  sxy /= N;
293  e_part = std::sqrt((sxx - syy)*(sxx - syy) + 4*sxy*sxy)/(sxx + syy);
294  }
295 
296 
297  // Need to calculate a few things missing from this list
298  HepMC::HeavyIon hi(1,np,nt,ncoll,0,0,n01,n10,n11,b,bphi,e_part,42.0);
299 
300  evt->set_heavy_ion (hi);
301 
302  // Set the random generator seeds. How do people handle the fact
303  // that CLHEP produced unsigned longs and HepMC wants signed ones?
304  // evt->set_random_states(engine->put());
305 
306  // Did we keep decay history?
307  bool keptHistory =(m_hiparnt.ihpr2(21) == 1);
308 
309  // The number of particles in the hijing output
310  int numHijingPart = m_himain1.natt();
311 
312  // Vectors that will keep track of where particles originate from and die
313  std::vector<int> partOriginVertex_vec;
314  std::vector<int> partDecayVertex_vec;
315  std::vector<HepMC::GenParticle *> particleHepPartPtr_vec;
316 
317  partOriginVertex_vec.assign(numHijingPart, 0);
318  partDecayVertex_vec.assign(numHijingPart, -1);
319  particleHepPartPtr_vec.assign(numHijingPart,(HepMC::GenParticle *) 0);
320 
321  // Vector that will keep pointers to generated vertices
322  std::vector<HepMC::GenVertex *> vertexPtrVec;
323 
324  // Create the event vertex
325  CLHEP::HepLorentzVector newVertex;
326  newVertex = CLHEP::HepLorentzVector(0., 0., 0., 0.);
327  HepMC::GenVertex *v1 = new HepMC::GenVertex(newVertex);
328 
329  evt->add_vertex(v1);
330  vertexPtrVec.push_back(v1);
331 
332  double eproj = (m_frame == "CMS") ? efrm / 2.0 : efrm;
333  int proj_id = (m_proj == "A") ? 3000000 + iap : 2212;
334  v1->add_particle_in(new HepMC::GenParticle(CLHEP::HepLorentzVector(0., 0., eproj, eproj),
335  proj_id, 101));
336 
337  double etarg = (m_frame == "CMS") ? efrm / 2.0 : efrm;
338  int targ_id = (m_targ == "A") ? 3000000 + iap : 2212;
339  v1->add_particle_in(new HepMC::GenParticle(CLHEP::HepLorentzVector(0., 0., -etarg, etarg),
340  targ_id, 102));
341 
342  // Loop on all Hijing particles and put them all as outgoing from the event vertex
343  for(int i = 1; i <= natt; ++i)
344  {
345  if (m_himain2.katt(i,4) == 103)
346  {
347  // We handle jets a little differently.
348  CLHEP::HepLorentzVector jetP4(m_himain2.patt(i,1),
349  m_himain2.patt(i,2),
350  m_himain2.patt(i,3),
351  m_himain2.patt(i,4));
352  v1->add_particle_out(new HepMC::GenParticle(jetP4,

354  m_himain2.katt(i,4)));
355 
356  continue;
357  }
358 
359  // Skip non-interacting projectile and target nucleons

382  // By default, the particle originates from the primary vertex
383  int particleVertexIndex = 0;
384 
385  // Have we kept the history?
386  if (keptHistory)

389  // for this parent
390  if(parentDecayIndex != -1)
391  {
392  // Make sure it is consistent with this track origin
393  HepPoint3D vertex_pos(vertexPtrVec[parentDecayIndex]->point3d().x(),
394  vertexPtrVec[parentDecayIndex]->point3d().y(),
395  vertexPtrVec[parentDecayIndex]->point3d().z());
396  double distance = vertex_pos.distance(particleStart.vect());
397 
398  if(distance > m_vertexOffsetCut)
399  {
400  HepMC::GenVertex::particles_out_const_iterator iter;
401  for (iter = vertexPtrVec[parentDecayIndex]->particles_out_const_begin();
402  iter != vertexPtrVec[parentDecayIndex]->particles_out_const_end();
403  iter++)
404  {
405  std::cout << (*iter)->barcode() << ", ";
406  }
407 
408  std::cout << std::endl;
409  }
410 
411  // Nonetheless set the parent decay vertex to be this particle's vertex
412  particleVertexIndex = parentDecayIndex;
413  }
414  else
415  {
416  // Now compare the distance between the vertex FROM
417  // which the parent originates and the start of this
418  // particle
419  HepPoint3D vertex_pos(vertexPtrVec[parentOriginIndex]->point3d().x(),
420  vertexPtrVec[parentOriginIndex]->point3d().y(),
421  vertexPtrVec[parentOriginIndex]->point3d().z());
422  double distance = vertex_pos.distance(particleStart.vect());
423 
424  if (distance > m_vertexOffsetCut && parentIndex == -1)
425  {
426  // *** Explicitly handle Hijing bug which generates
427  // *** particle with displaced vertex but no parent
428  // *** ????
429 
430  std::cout

434  if (distance > m_vertexOffsetCut || parentOriginIndex != 0)
435  {
436  // We need to create a new vertex
437  HepMC::GenVertex * newVertex_p = new HepMC::GenVertex(particleStart);
438 
439  evt->add_vertex(newVertex_p);
440  vertexPtrVec.push_back(newVertex_p);
441  particleVertexIndex = vertexPtrVec.size() - 1;
442 
443  // Now we indicate that the parent has a decay vertex
444  partDecayVertex_vec[parentIndex] = particleVertexIndex;
445 
446  // Now tell the vertex about the particle that created it
447  newVertex_p->add_particle_in(particleHepPartPtr_vec[parentIndex]);
448  }

470  }
471  }
472 
473  if (foundVert == -1)
474  {
475  // We need to create a new vertex
476  HepMC::GenVertex *newVertex_p = new HepMC::GenVertex(particleStart);
477 
478  evt->add_vertex(newVertex_p);
479  vertexPtrVec.push_back(newVertex_p);
480  particleVertexIndex = vertexPtrVec.size() - 1;
481  }
482  else
483  {
484  particleVertexIndex = foundVert;
485  }
486  }
487 
488  // If the Hijing particle has decayed, set the status appropriately
489  int particleId = m_himain2.katt(i, 1);
490  int particleStatus = 1;
491  if (m_himain2.katt(i, 4) == 11)
492  {
493  particleStatus = 2;
494  }
495 
496  // Create the new particle
497  CLHEP::HepLorentzVector particleP4(m_himain2.patt(i, 1),
498  m_himain2.patt(i, 2),
499  m_himain2.patt(i, 3),
500  m_himain2.patt(i, 4));
501 
502  HepMC::GenParticle *newParticle_p = new HepMC::GenParticle(particleP4,
503  particleId,
504  particleStatus);
505 
506  // Record the particle in the vector of pointers (ostensibly we
507  // only need this when we have the history but for simplicity
508  // always do it)
509  particleHepPartPtr_vec[i - 1] = newParticle_p;
510 
511  // Now add the particle to its vertex
512  vertexPtrVec[particleVertexIndex]->add_particle_out(newParticle_p);
513  partOriginVertex_vec[i - 1] = particleVertexIndex;
514  }
515
