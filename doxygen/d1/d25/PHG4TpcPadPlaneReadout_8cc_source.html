511  double radius = LayerGeom->get_radius();
512  double phidisp = phi - LayerGeom->get_phicenter(phibin);
513  double phistepsize = LayerGeom->get_phistep();
514 
515  // bin the charge in phi - consider phi bins up and down 3 sigma in r-phi
516  int n_rp = int(3 * cloud_sig_rp / (radius * phistepsize) + 1);
517  for (int iphi = -n_rp; iphi != n_rp + 1; ++iphi)
518  {
519  int cur_phi_bin = phibin + iphi;
520  // correcting for continuity in phi
521  if (cur_phi_bin < 0)
522  cur_phi_bin += nphibins;
523  else if (cur_phi_bin >= nphibins)
524  cur_phi_bin -= nphibins;
525  if ((cur_phi_bin < 0) || (cur_phi_bin >= nphibins))
526  {
527  std::cout << "PHG4CylinderCellTpcReco => error in phi continuity. Skipping" << std::endl;
528  continue;
529  }
530  // Get the integral of the charge probability distribution in phi inside the current phi step
531  double phiLim1 = 0.5 * M_SQRT2 * ((iphi + 0.5) * phistepsize * radius - phidisp * radius) * cloud_sig_rp_inv;
532  double phiLim2 = 0.5 * M_SQRT2 * ((iphi - 0.5) * phistepsize * radius - phidisp * radius) * cloud_sig_rp_inv;
533  double phi_integral = 0.5 * (erf(phiLim1) - erf(phiLim2));
534 
535  pad_phibin.push_back(cur_phi_bin);
536  pad_phibin_share.push_back(phi_integral);
537  }
538 
539  return;
540 }

542 void PHG4TpcPadPlaneReadout::populate_zigzag_phibins(const unsigned int layernum, const double phi, const double cloud_sig_rp, std::vector<int> &pad_phibin, std::vector<double> &pad_phibin_share)
543 {
544  double nsigmas = 5.0;
545 
546  double radius = LayerGeom->get_radius();
547  double phistepsize = LayerGeom->get_phistep();
548 
549  // make the charge distribution gaussian
550  double rphi = phi * radius;
551  fcharge->SetParameter(0, 1.0);
552  fcharge->SetParameter(1, rphi);
553  fcharge->SetParameter(2, cloud_sig_rp);
554  if (Verbosity() > 100)
555  if (LayerGeom->get_layer() == 47)
556  {
557  cout << " populate_zigzag_phibins for layer " << layernum << " with radius " << radius << " phi " << phi
558  << " rphi " << rphi << " phistepsize " << phistepsize << endl;
559  cout << " fcharge created: radius " << radius << " rphi " << rphi << " cloud_sig_rp " << cloud_sig_rp << endl;
560  }
561 
562  // Get the range of phi values that completely contains all pads that touch the charge distribution - (nsigmas + 1/2 pad width) in each direction
563  double philim_low = phi - (nsigmas * cloud_sig_rp / radius) - phistepsize;
564  double philim_high = phi + (nsigmas * cloud_sig_rp / radius) + phistepsize;
565 
566  // Find the pad range that covers this phi range
567  int phibin_low = LayerGeom->get_phibin(philim_low);
568  int phibin_high = LayerGeom->get_phibin(philim_high);
569  int npads = phibin_high - phibin_low;
570 
571  if (Verbosity() > 100)
572  if (layernum == 47)
573  cout << " zigzags: phi " << phi << " philim_low " << philim_low << " phibin_low " << phibin_low
574  << " philim_high " << philim_high << " phibin_high " << phibin_high << " npads " << npads << endl;
575 
576  if (npads < 0 || npads > 9) npads = 9; // can happen if phibin_high wraps around. If so, limit to 10 pads and fix below
577 
578  // Calculate the maximum extent in r-phi of pads in this layer. Pads are assumed to touch the center of the next phi bin on both sides.
579  double pad_rphi = 2.0 * LayerGeom->get_phistep() * radius;
580 
581  // Make a TF1 for each pad in the phi range
582  int pad_keep[10];
583  for (int ipad = 0; ipad <= npads; ipad++)
584  {
585  int pad_now = phibin_low + ipad;
586  // check that we do not exceed the maximum number of pads, wrap if necessary
587  if (pad_now >= LayerGeom->get_phibins()) pad_now -= LayerGeom->get_phibins();
588 
589  pad_keep[ipad] = pad_now;
590  double rphi_pad_now = LayerGeom->get_phicenter(pad_now) * radius;
591 
592  fpad[ipad]->SetParameter(0, pad_rphi / 2.0);
593  fpad[ipad]->SetParameter(1, rphi_pad_now);
594 
595  if (Verbosity() > 100)
596  if (layernum == 47)
597  cout << " zigzags: make fpad for ipad " << ipad << " pad_now " << pad_now << " pad_rphi/2 " << pad_rphi / 2.0
598  << " rphi_pad_now " << rphi_pad_now << endl;
599  }
600 
601  // Now make a loop that steps through the charge distribution and evaluates the response at that point on each pad
602 
603  double overlap[10];
604  for (int i = 0; i < 10; i++)
605  overlap[i] = 0;
606 
607  int nsteps = 100;
608  double xstep = 2.0 * nsigmas * cloud_sig_rp / (double) nsteps;
609  for (int i = 0; i < nsteps; i++)
610  {
611  double x = rphi - 4.5 * cloud_sig_rp + (double) i * xstep;
612  double charge = fcharge->Eval(x);
613  for (int ipad = 0; ipad <= npads; ipad++)
614  {
615  if (fpad[ipad]->Eval(x) > 0.0)
616  {
617  double prod = charge * fpad[ipad]->Eval(x);
618  overlap[ipad] += prod;
619  }
620  } // pads
621 
622  } // steps
623 
624  // now we have the overlap for each pad
625  for (int ipad = 0; ipad <= npads; ipad++)
626  {
627  pad_phibin.push_back(pad_keep[ipad]);
628  pad_phibin_share.push_back(overlap[ipad]);
629  if (rad_gem < output_radius) cout << " zigzags: for pad " << ipad << " integral is " << overlap[ipad] << endl;
630  }
631 
632  return;
633 }

635 void PHG4TpcPadPlaneReadout::populate_zbins(const double z, const double cloud_sig_zz[2], std::vector<int> &adc_zbin, std::vector<double> &adc_zbin_share)
636 {
637  int zbin = LayerGeom->get_zbin(z);
638  if (zbin < 0 || zbin > LayerGeom->get_zbins())
639  {
640  //cout << " z bin is outside range, return" << endl;
641  return;
642  }
643 
644  double zstepsize = LayerGeom->get_zstep();
645  double zdisp = z - LayerGeom->get_zcenter(zbin);
646 
647  if (Verbosity() > 100)
648  cout << " input: z " << z << " zbin " << zbin << " zstepsize " << zstepsize << " z center " << LayerGeom->get_zcenter(zbin) << " zdisp " << zdisp << endl;
649 
650  // Because of diffusion, hits can be shared across the membrane, so we allow all z bins
651  int min_cell_zbin = 0;
652  int max_cell_zbin = NZBins - 1;
653 
654  double cloud_sig_zz_inv[2];
655  cloud_sig_zz_inv[0] = 1. / cloud_sig_zz[0];
656  cloud_sig_zz_inv[1] = 1. / cloud_sig_zz[1];
657 
658  int zsect = 0;
659  if (z < 0)
660  zsect = -1;
661  else
662  zsect = 1;
663 
664  int n_zz = int(3 * (cloud_sig_zz[0] + cloud_sig_zz[1]) / (2.0 * zstepsize) + 1);
665  if (Verbosity() > 100) cout << " n_zz " << n_zz << " cloud_sigzz[0] " << cloud_sig_zz[0] << " cloud_sig_zz[1] " << cloud_sig_zz[1] << endl;
666  for (int iz = -n_zz; iz != n_zz + 1; ++iz)
667  {
668  int cur_z_bin = zbin + iz;
669  if ((cur_z_bin < min_cell_zbin) || (cur_z_bin > max_cell_zbin)) continue;
670 
671  if (Verbosity() > 100)
672  cout << " iz " << iz << " cur_z_bin " << cur_z_bin << " min_cell_zbin " << min_cell_zbin << " max_cell_zbin " << max_cell_zbin << endl;
673 
674  double z_integral = 0.0;
675  if (iz == 0)
676  {
677  // the crossover between lead and tail shaping occurs in this bin
678  int index1 = -1;
679  int index2 = -1;
680  if (zsect == -1)
681  {
682  index1 = 0;
683  index2 = 1;
684  }
685  else
686  {
687  index1 = 1;
688  index2 = 0;
689  }
690 
691  double zLim1 = 0.0;
692  double zLim2 = 0.5 * M_SQRT2 * (-0.5 * zstepsize - zdisp) * cloud_sig_zz_inv[index1];
693  // 1/2 * the erf is the integral probability from the argument Z value to zero, so this is the integral probability between the Z limits
694  double z_integral1 = 0.5 * (erf(zLim1) - erf(zLim2));
695 
696  if (Verbosity() > 100)
697  if (LayerGeom->get_layer() == 47)
698  cout << " populate_zbins: cur_z_bin " << cur_z_bin << " center z " << LayerGeom->get_zcenter(cur_z_bin)
699  << " index1 " << index1 << " zLim1 " << zLim1 << " zLim2 " << zLim2 << " z_integral1 " << z_integral1 << endl;
700 
701  zLim2 = 0.0;
702  zLim1 = 0.5 * M_SQRT2 * (0.5 * zstepsize - zdisp) * cloud_sig_zz_inv[index2];
703  double z_integral2 = 0.5 * (erf(zLim1) - erf(zLim2));
704 
705  if (Verbosity() > 100)
706  if (LayerGeom->get_layer() == 47)
707  cout << " populate_zbins: cur_z_bin " << cur_z_bin << " center z " << LayerGeom->get_zcenter(cur_z_bin)
708  << " index2 " << index2 << " zLim1 " << zLim1 << " zLim2 " << zLim2 << " z_integral2 " << z_integral2 << endl;
709 
710  z_integral = z_integral1 + z_integral2;
711  }
712  else
713  {
714  // The non zero bins are entirely in the lead or tail region
715  // lead or tail depends on which side of the membrane
716  int index = 0;
717  if (iz < 0)
718  {
719  if (zsect == -1)
720  index = 0;
721  else
722  index = 1;
723  }
724  else
725  {
726  if (zsect == -1)
727  index = 1;
728  else
729  index = 0;
730  }
731  double zLim1 = 0.5 * M_SQRT2 * ((iz + 0.5) * zstepsize - zdisp) * cloud_sig_zz_inv[index];
732  double zLim2 = 0.5 * M_SQRT2 * ((iz - 0.5) * zstepsize - zdisp) * cloud_sig_zz_inv[index];
733  z_integral = 0.5 * (erf(zLim1) - erf(zLim2));
734 
735  if (Verbosity() > 100)
736  if (LayerGeom->get_layer() == 47)
737  cout << " populate_zbins: z_bin " << cur_z_bin << " center z " << LayerGeom->get_zcenter(cur_z_bin)
738  << " index " << index << " zLim1 " << zLim1 << " zLim2 " << zLim2 << " z_integral " << z_integral << endl;
739  }
740 
741  adc_zbin.push_back(cur_z_bin);
742  adc_zbin_share.push_back(z_integral);
743  }
744 
745  return;
746 }
