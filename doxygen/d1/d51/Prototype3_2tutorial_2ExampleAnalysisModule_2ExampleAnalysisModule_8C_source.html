537 pair<int, int>
538 ExampleAnalysisModule::find_max(RawTowerContainer* towers, int cluster_size)
539 {
540  const int clus_edge_size = (cluster_size - 1) / 2;
541  assert(clus_edge_size >= 0);
542 
543  pair<int, int> max(-1000, -1000);
544  double max_e = 0;
545 
546  for (int col = 0; col < n_size; ++col)
547  for (int row = 0; row < n_size; ++row)
548  {
549  double energy = 0;
550 
551  for (int dcol = col - clus_edge_size; dcol <= col + clus_edge_size;
552  ++dcol)
553  for (int drow = row - clus_edge_size; drow <= row + clus_edge_size;
554  ++drow)
555  {
556  if (dcol < 0 or drow < 0)
557  continue;
558 
559  RawTower * t = towers->getTower(dcol, drow);
560  if (t)
561  energy += t->get_energy();
562  }
563 
564  if (energy > max_e)
565  {
566  max = make_pair(col, row);
567  max_e = energy;
568  }
569  }
570 
571  return max;
572 }
