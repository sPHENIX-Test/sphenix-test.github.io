336 void RawTowerDigitizer::CreateNodes(PHCompositeNode *topNode)
337 {

339  PHCompositeNode *runNode = static_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "RUN"));
340  if (!runNode)
341  {
342  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
343  << "Run Node missing, doing nothing." << endl;
344  throw std::runtime_error(
345  "Failed to find Run node in RawTowerDigitizer::CreateNodes");
346  }
347 
348  string TowerGeomNodeName = "TOWERGEOM_" + m_Detector;
349  m_RawTowerGeom = findNode::getClass<RawTowerGeomContainer>(topNode, TowerGeomNodeName);
350  if (!m_RawTowerGeom)
351  {
352  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
353  << " " << TowerGeomNodeName << " Node missing, doing bail out!"
354  << endl;
355  throw std::runtime_error("Failed to find " + TowerGeomNodeName + " node in RawTowerDigitizer::CreateNodes");
356  }
357 
358  if (Verbosity() >= 1)
359  {
360  m_RawTowerGeom->identify();
361  }
362 
363  const string deadMapName = "DEADMAP_" + m_Detector;
364  m_DeadMap = findNode::getClass<RawTowerDeadMap>(topNode, deadMapName);
365  if (m_DeadMap)
366  {
367  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
368  << " use dead map: ";
369  m_DeadMap->identify();
370  }
371 
372  PHCompositeNode *dstNode = dynamic_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "DST"));
373  if (!dstNode)
374  {
375  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
376  << "DST Node missing, doing nothing." << endl;
377  throw std::runtime_error("Failed to find DST node in RawTowerDigitizer::CreateNodes");
378  }
379 
380  string SimTowerNodeName = "TOWER_" + m_SimTowerNodePrefix + "_" + m_Detector;
381  m_SimTowers = findNode::getClass<RawTowerContainer>(dstNode, SimTowerNodeName);
382  if (!m_SimTowers)
383  {
384  cout << Name() << "::" << m_Detector << "::" << __PRETTY_FUNCTION__
385  << " " << SimTowerNodeName << " Node missing, doing bail out!"
386  << endl;
387  throw std::runtime_error("Failed to find " + SimTowerNodeName + " node in RawTowerDigitizer::CreateNodes");
388  }
389 
390  // Create the tower nodes on the tree
391  PHNodeIterator dstiter(dstNode);
392  PHCompositeNode *DetNode = dynamic_cast<PHCompositeNode *>(dstiter.findFirst("PHCompositeNode", m_Detector));
393  if (!DetNode)
394  {
395  DetNode = new PHCompositeNode(m_Detector);
396  dstNode->addNode(DetNode);
397  }
398 
399  // Be careful as a previous digitizer may have been registered for this detector
400  string RawTowerNodeName = "TOWER_" + m_RawTowerNodePrefix + "_" + m_Detector;
401  m_RawTowers = findNode::getClass<RawTowerContainer>(DetNode, RawTowerNodeName);
402  if (!m_RawTowers)
403  {
404  m_RawTowers = new RawTowerContainer(m_SimTowers->getCalorimeterID());
405  PHIODataNode<PHObject> *towerNode = new PHIODataNode<PHObject>(m_RawTowers,
406  RawTowerNodeName, "PHObject");

408  }
409
