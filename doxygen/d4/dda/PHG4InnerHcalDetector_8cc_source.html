434  // e.g. av_1_impr_82_HcalInnerScinti_11_pv_11
435  // 82 the number of the scintillator mother volume
436  // HcalInnerScinti_11: name of scintillator slat
437  // 11: number of scintillator slat logical volume
438  // use boost tokenizer to separate the _, then take value
439  // after "impr" for mother volume and after "pv" for scintillator slat
440  // use boost lexical cast for string -> int conversion
441  // the CopyNo is the mother volume + scinti id
442  // so we can use the CopyNo rather than decoding the string further
443  // looking for "pv"
444  boost::char_separator<char> sep("_");
445  boost::tokenizer<boost::char_separator<char>> tok((*it)->GetName(), sep);
446  boost::tokenizer<boost::char_separator<char>>::const_iterator tokeniter;
447  for (tokeniter = tok.begin(); tokeniter != tok.end(); ++tokeniter)
448  {
449  if (*tokeniter == "impr")
450  {
451  ++tokeniter;
452  if (tokeniter != tok.end())
453  {
454  int layer_id = boost::lexical_cast<int>(*tokeniter);
455  // check detector description, for assemblyvolumes it is not possible
456  // to give the first volume id=0, so they go from id=1 to id=n.
457  // I am not going to start with fortran again - our indices start
458  // at zero, id=0 to id=n-1. So subtract one here
459  int tower_id = (*it)->GetCopyNo() - layer_id;
460  layer_id--;
461  pair<int, int> layer_twr = make_pair(layer_id, tower_id);
462  m_ScintiTilePhysVolMap.insert(pair<G4VPhysicalVolume *, pair<int, int>>(*it, layer_twr));
463  if (layer_id < 0 || layer_id >= m_NumScintiPlates)
464  {
465  cout << "invalid scintillator row " << layer_id
466  << ", valid range 0 < row < " << m_NumScintiPlates << endl;
467  gSystem->Exit(1);
468  }
469  }
470  else
471  {
472  cout << PHWHERE << " Error parsing " << (*it)->GetName()
473  << " for mother volume number " << endl;
474  gSystem->Exit(1);
475  }
476  break;
477  }
478  }
479  ++it;
480  }
481  return;

484 int PHG4InnerHcalDetector::ConstructInnerHcal(G4LogicalVolume *hcalenvelope)
485 {
486  ConsistencyCheck();
487  SetTiltViaNcross(); // if number of crossings is set, use it to determine tilt
488  CheckTiltAngle(); // die if the tilt angle is out of range
489  G4VSolid *steel_plate = ConstructSteelPlate(hcalenvelope);
490  G4LogicalVolume *steel_logical = new G4LogicalVolume(steel_plate, G4Material::GetMaterial(m_Params->get_string_param("material")), "HcalInnerSteelPlate", 0, 0, 0);
491  m_DisplayAction->AddSteelVolume(steel_logical);
492  m_ScintiMotherAssembly = ConstructHcalScintillatorAssembly(hcalenvelope);
493  double phi = 0;
494  double deltaphi = 2 * M_PI / m_NumScintiPlates;
495  ostringstream name;
496  double middlerad = m_OuterRadius - (m_OuterRadius - m_InnerRadius) / 2.;
497  // for shorter scintillators we have to shift by 1/2 the "missing length"
498  double scinti_tile_orig_length = m_ScintiTileXUpper + m_ScintiTileXLower - subtract_from_scinti_x;
499  double shiftslat = fabs(m_ScintiTileXLower - m_ScintiTileXUpper) / 2. + (scinti_tile_orig_length - m_ScintiTileX) / 2.;
500  // calculate phi offset (copied from code inside following loop):
501  // first get the center point (phi=0) so it's middlerad/0
502  // then shift the scintillator center as documented in loop
503  // then
504  // for positive tilt angles we need the lower left corner of the scintillator
505  // for negative tilt angles we need the upper right corner of the scintillator
506  // as it turns out the code uses the middle of the face of the scintillator
507  // as reference, if this is a problem the code needs to be modified to
508  // actually calculate the corner (but the math of the construction is that
509  // the middle of the scintillator sits at zero)
510  double xp = cos(phi) * middlerad;
511  double yp = sin(phi) * middlerad;
512  xp -= cos((-m_TiltAngle) / rad - phi) * shiftslat;
513  yp += sin((-m_TiltAngle) / rad - phi) * shiftslat;
514  if (m_TiltAngle > 0)
515  {
516  double xo = xp - (m_ScintiTileX / 2.) * cos(m_TiltAngle / rad);
517  double yo = yp - (m_ScintiTileX / 2.) * sin(m_TiltAngle / rad);
518  phi = -atan(yo / xo);
519  }
520  else if (m_TiltAngle < 0)
521  {
522  double xo = xp + (m_ScintiTileX / 2.) * cos(m_TiltAngle / rad);
523  double yo = yp + (m_ScintiTileX / 2.) * sin(m_TiltAngle / rad);
524  phi = -atan(yo / xo);
525  }
526  // else (for m_TiltAngle = 0) phi stays zero
527  for (int i = 0; i < m_NumScintiPlates; i++)
528  {
529  G4RotationMatrix *Rot = new G4RotationMatrix();
530  double ypos = sin(phi) * middlerad;
531  double xpos = cos(phi) * middlerad;
532  // the center of the scintillator is not the center of the inner hcal
533  // but depends on the tilt angle. Therefore we need to shift
534  // the center from the mid point
535  ypos += sin((-m_TiltAngle) / rad - phi) * shiftslat;
536  xpos -= cos((-m_TiltAngle) / rad - phi) * shiftslat;
537  Rot->rotateZ(phi * rad + m_TiltAngle);
538  G4ThreeVector g4vec(xpos, ypos, 0);
539  // great the MakeImprint always adds 1 to the copy number and 0 has a
540  // special meaning (which then also adds 1). Basically our volume names
541  // will start at 1 instead of 0 and there is nothing short of patching this
542  // method. I'll take care of this in the decoding of the volume name
543  // AAAAAAARHGS
544  m_ScintiMotherAssembly->MakeImprint(hcalenvelope, g4vec, Rot, i, OverlapCheck());
545  delete Rot;
546  Rot = new G4RotationMatrix();
547  Rot->rotateZ(-phi * rad);
548  name.str("");
549  name << "InnerHcalSteel_" << i;
550  m_SteelAbsorberPhysVolSet.insert(new G4PVPlacement(Rot, G4ThreeVector(0, 0, 0), steel_logical, name.str().c_str(), hcalenvelope, 0, i, OverlapCheck()));
551  phi += deltaphi;
552  }
553  return 0;
554 }

559 void PHG4InnerHcalDetector::ConstructHcalSingleScintillators(G4LogicalVolume *hcalenvelope)
560 {
561  G4VSolid *bigtile = ConstructScintillatorBox(hcalenvelope);
562  // eta->theta
563  G4double delta_eta = m_Params->get_double_param("scinti_eta_coverage") / m_NumScintiTiles;
564  G4double eta = 0;
565  G4double theta;
566  G4double x[4];
567  G4double z[4];
568  ostringstream name;
569  double overhang = (m_ScintiTileX - (m_OuterRadius - m_InnerRadius)) / 2.;
570  double offset = 1 * cm + overhang; // add 1cm to make sure the G4ExtrudedSolid
571  // is larger than the tile so we do not have
572  // funny edge effects when overlapping vols
573  double scinti_gap_neighbor = m_Params->get_double_param("scinti_gap_neighbor") * cm;
574  for (int i = 0; i < m_NumScintiTiles; i++)
575  {
576  theta = M_PI / 2 - PHG4Utils::get_theta(eta); // theta = 90 for eta=0
577  x[0] = m_InnerRadius - overhang;
578  z[0] = tan(theta) * m_InnerRadius;
579  x[1] = m_OuterRadius + overhang; // since the tile is tilted, x is not at the outer radius but beyond
580  z[1] = tan(theta) * m_OuterRadius;
581  eta += delta_eta;
582  theta = M_PI / 2 - PHG4Utils::get_theta(eta); // theta = 90 for eta=0
583  x[2] = m_InnerRadius - overhang;
584  z[2] = tan(theta) * m_InnerRadius;
585  x[3] = m_OuterRadius + overhang; // since the tile is tilted, x is not at the outer radius but beyond
586  z[3] = tan(theta) * m_OuterRadius;
587  // apply gap between scintillators
588  z[0] += scinti_gap_neighbor / 2.;
589  z[1] += scinti_gap_neighbor / 2.;
590  z[2] -= scinti_gap_neighbor / 2.;
591  z[3] -= scinti_gap_neighbor / 2.;
592  Point_2 leftsidelow(z[0], x[0]);
593  Point_2 leftsidehigh(z[1], x[1]);
594  x[0] = m_InnerRadius - offset;
595  z[0] = x_at_y(leftsidelow, leftsidehigh, x[0]);
596  x[1] = m_OuterRadius + offset;
597  z[1] = x_at_y(leftsidelow, leftsidehigh, x[1]);
598  Point_2 rightsidelow(z[2], x[2]);
599  Point_2 rightsidehigh(z[3], x[3]);
600  x[2] = m_OuterRadius + offset;
601  z[2] = x_at_y(rightsidelow, rightsidehigh, x[2]);
602  x[3] = m_InnerRadius - offset;
603  z[3] = x_at_y(rightsidelow, rightsidehigh, x[3]);
604 
605  vector<G4TwoVector> vertexes;
606  for (int j = 0; j < 4; j++)
607  {
608  G4TwoVector v(x[j], z[j]);
609  vertexes.push_back(v);
610  }
611  G4TwoVector zero(0, 0);
612 
613  G4VSolid *scinti = new G4ExtrudedSolid("ScintillatorTile",
614  vertexes,
615  m_ScintiTileThickness + 0.2 * mm,
616  zero, 1.0,
617  zero, 1.0);
618  G4RotationMatrix *rotm = new G4RotationMatrix();
619  rotm->rotateX(-90 * deg);
620  name.str("");
621  name << "scintillator_" << i << "_left";
622  G4VSolid *scinti_tile = new G4IntersectionSolid(name.str(), bigtile, scinti, rotm, G4ThreeVector(-(m_InnerRadius + m_OuterRadius) / 2., 0, 0));
623  delete rotm;
624  m_ScintiTilesVec[i + m_NumScintiTiles] = scinti_tile;
625  rotm = new G4RotationMatrix();
626  rotm->rotateX(90 * deg);
627 
628  name.str("");
629  name << "scintillator_" << i << "_right";
630  scinti_tile = new G4IntersectionSolid(name.str(), bigtile, scinti, rotm, G4ThreeVector(-(m_InnerRadius + m_OuterRadius) / 2., 0, 0));
631  m_ScintiTilesVec[m_NumScintiTiles - i - 1] = scinti_tile;
632  delete rotm;
633  }

702 int PHG4InnerHcalDetector::CheckTiltAngle() const
703 {
704  if (fabs(m_TiltAngle / rad) >= M_PI)
705  {
706  cout << PHWHERE << "invalid tilt angle, abs(tilt) >= 90 deg: " << (m_TiltAngle / deg)
707  << endl;
708  exit(1);
709  }
710 
711  double mid_radius = m_InnerRadius + (m_OuterRadius - m_InnerRadius) / 2.;
712  Point_2 pmid(mid_radius, 0); // center of scintillator
713  double xcoord = 0;
714  double ycoord = mid_radius * tan(m_TiltAngle / rad);
715  Point_2 pxnull(xcoord, ycoord);
716  Line_2 s2(pmid, pxnull);
717  Point_2 sc1(m_InnerRadius, 0), sc2(0, m_InnerRadius), sc3(-m_InnerRadius, 0);
718  Circle_2 inner_circle(sc1, sc2, sc3);
719  vector<CGAL::Object> res;
720  CGAL::intersection(inner_circle, s2, std::back_inserter(res));
721  if (res.size() == 0)

725  exit(1);

757 void PHG4InnerHcalDetector::SetTiltViaNcross()
758 {
759  // if tilt angle is set it takes precedence
760  // this happens during readback where the tilt angle was set
761  // in this method
762  int ncross = m_Params->get_int_param("ncross");
763  if (!ncross || isfinite(m_TiltAngle))
764  {
765  // flag ncrossing as not used
766  m_Params->set_int_param("ncross", 0);
767  return;
768  }
769  if ((isfinite(m_TiltAngle)) && (Verbosity() > 0))
770  {
771  cout << "both number of crossings and tilt angle are set" << endl;
772  cout << "using number of crossings to determine tilt angle" << endl;
773  }
774  double mid_radius = m_InnerRadius + (m_OuterRadius - m_InnerRadius) / 2.;
775  double deltaphi = (2 * M_PI / m_NumScintiPlates) * ncross;
776  Point_2 pnull(0, 0);
777  Point_2 plow(m_InnerRadius, 0);
778  Point_2 phightmp(1, tan(deltaphi));
779  Point_2 pin1(m_InnerRadius, 0), pin2(0, m_InnerRadius), pin3(-m_InnerRadius, 0);
780  Circle_2 inner_circle(pin1, pin2, pin3);
781  Point_2 pmid1(mid_radius, 0), pmid2(0, mid_radius), pmid3(-mid_radius, 0);
782  Circle_2 mid_circle(pmid1, pmid2, pmid3);
783  Point_2 pout1(m_OuterRadius, 0), pout2(0, m_OuterRadius), pout3(-m_OuterRadius, 0);
784  Circle_2 outer_circle(pout1, pout2, pout3);
785  Line_2 l_up(pnull, phightmp);
786  vector<CGAL::Object> res;
787  CGAL::intersection(outer_circle, l_up, std::back_inserter(res));
788  Point_2 upperright;
789  vector<CGAL::Object>::const_iterator iter;
790  for (iter = res.begin(); iter != res.end(); ++iter)
791  {
792  CGAL::Object obj = *iter;
793  if (const std::pair<CGAL::Circular_arc_point_2<PHG4InnerHcalDetector::Circular_k>, unsigned> *point = CGAL::object_cast<std::pair<CGAL::Circular_arc_point_2<PHG4InnerHcalDetector::Circular_k>, unsigned>>(&obj))
794  {
795  if (CGAL::to_double(point->first.x()) > 0)
796  {
797  Point_2 pntmp(CGAL::to_double(point->first.x()), CGAL::to_double(point->first.y()));
798  upperright = pntmp;
799  }
800  }
801  else
802  {
803  cout << "CGAL::Object type not pair..." << endl;
804  exit(1);
805  }
806  }
807  Line_2 l_right(plow, upperright);
808  res.clear();
809  Point_2 midpoint;
810  CGAL::intersection(mid_circle, l_right, std::back_inserter(res));
811  for (iter = res.begin(); iter != res.end(); ++iter)
812  {
813  CGAL::Object obj = *iter;
814  if (const std::pair<CGAL::Circular_arc_point_2<PHG4InnerHcalDetector::Circular_k>, unsigned> *point = CGAL::object_cast<std::pair<CGAL::Circular_arc_point_2<PHG4InnerHcalDetector::Circular_k>, unsigned>>(&obj))
815  {
816  if (CGAL::to_double(point->first.x()) > 0)
817  {
818  Point_2 pntmp(CGAL::to_double(point->first.x()), CGAL::to_double(point->first.y()));
819  midpoint = pntmp;
820  }
821  }
822  else
823  {
824  cout << "CGAL::Object type not pair..." << endl;
825  exit(1);
826  }
827  }
828  // length left side
829  double ll = sqrt((CGAL::to_double(midpoint.x()) - m_InnerRadius) * (CGAL::to_double(midpoint.x()) - m_InnerRadius) + CGAL::to_double(midpoint.y()) * CGAL::to_double(midpoint.y()));
830  double upside = sqrt(CGAL::to_double(midpoint.x()) * CGAL::to_double(midpoint.x()) + CGAL::to_double(midpoint.y()) * CGAL::to_double(midpoint.y()));
831  // c^2 = a^2+b^2 - 2ab*cos(gamma)
832  // gamma = acos((a^2+b^2=c^2)/2ab
833  double tiltangle = acos((ll * ll + upside * upside - m_InnerRadius * m_InnerRadius) / (2 * ll * upside));
834  tiltangle = tiltangle * rad;
835  m_TiltAngle = copysign(tiltangle, ncross);
836  m_Params->set_double_param("tilt_angle", m_TiltAngle / deg);
837  return;
838 }
