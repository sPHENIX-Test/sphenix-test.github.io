56 void PHG4RICHSteppingAction::UserSteppingAction(const G4Step* aStep)
57 {
58  G4Track* theTrack = aStep->GetTrack();
59 
60  G4StepPoint* thePostPoint = aStep->GetPostStepPoint();
61  G4VPhysicalVolume* thePostPV = thePostPoint->GetPhysicalVolume();
62 
63  G4OpBoundaryProcessStatus boundaryStatus = Undefined;
64  static G4OpBoundaryProcess* boundary = nullptr;
65 
66  /* find the boundary process only once */
67  if (!boundary)
68  {
69  G4ProcessManager* pm = aStep->GetTrack()->GetDefinition()->GetProcessManager();
70  G4int nprocesses = pm->GetProcessListLength();
71  G4ProcessVector* pv = pm->GetProcessList();
72  G4int i;
73  for (i = 0; i < nprocesses; i++)
74  {
75  if ((*pv)[i]->GetProcessName() == "OpBoundary")
76  {
77  boundary = (G4OpBoundaryProcess*) (*pv)[i];
78  break;
79  }
80  }
81  }
82 
83  if (!thePostPV)
84  { //out of world
85  return;
86  }
87 
88  /* Optical photon only */
89  G4ParticleDefinition* particleType = theTrack->GetDefinition();
90  if (particleType == G4OpticalPhoton::OpticalPhotonDefinition())
91  {
92  /* Was the photon absorbed by the absorption process? */
93  if (thePostPoint->GetProcessDefinedStep()->GetProcessName() == "OpAbsorption")
94  {
95  }
96  assert(boundary != nullptr);
97  boundaryStatus = boundary->GetStatus();
98 
99  /*Check to see if the partcile was actually at a boundary
100  Otherwise the boundary status may not be valid
101  Prior to Geant4.6.0-p1 this would not have been enough to check */
102  if (thePostPoint->GetStepStatus() == fGeomBoundary)
103  {
104  if (fExpectedNextStatus == StepTooSmall)
105  {
106  if (boundaryStatus != StepTooSmall)
107  {
108  G4ExceptionDescription ed;
109  ed << "EicRichGemTbSteppingAction::UserSteppingAction(): "
110  << "No reallocation step after reflection!"
111  << G4endl;
112  G4Exception("EicRichGemTbSteppingAction::UserSteppingAction()", "EicRichGemTbExpl01",
113  FatalException, ed,
114  "Something is wrong with the surface normal or geometry");
115  }
116  }
117  fExpectedNextStatus = Undefined;
118  switch (boundaryStatus)
119  {
120  case Absorption:
121  break;
122  case Detection: /*Note, this assumes that the volume causing detection
123  is the photocathode because it is the only one with
124  non-zero efficiency */
125  {
126  /* make sure the photon actually did hit the GEM stack */
127  if (thePostPV->GetName() == "RICHHBDGEMFrontCu1Physical")
128  {
129  MakeHit(aStep);
130  }
131  break;
132  }
133  case FresnelReflection:
134  case TotalInternalReflection:
135  case LambertianReflection:
136  case LobeReflection:
137  case SpikeReflection:
138  case BackScattering:
139  fExpectedNextStatus = StepTooSmall;
140  break;
141  default:
142  break;
143  }
144  }
145  }
146 
147  return;

150 bool PHG4RICHSteppingAction::MakeHit(const G4Step* aStep)
151 {
152  // collect energy and track
153  G4double edep = aStep->GetTotalEnergyDeposit() / GeV;
154  const G4Track* aTrack = aStep->GetTrack();
155  const G4VTouchable* aTouch = aTrack->GetTouchable();
156  G4StepPoint* postPoint = aStep->GetPostStepPoint();
157 
158  // set sector number
159  int sector_id = -999;
160  bool sector_found = false;
161 
162  // Check if volume(1) is in sector volume, if not check volume(0)
163  if (detector_->ePHENIXRICHConstruction::is_in_sector(aTouch->GetVolume(1)) > -1)
164  {
165  sector_id = aTouch->GetCopyNumber(1);
166  sector_found = true;
167  }
168  else if (detector_->ePHENIXRICHConstruction::is_in_sector(aTouch->GetVolume()) > -1)
169  {
170  sector_id = aTouch->GetCopyNumber();
171  sector_found = true;
172  }
173 
174  if (!sector_found)
175  {
176  if (!aTouch->GetVolume(1) || !aTouch->GetVolume())
177  cout << "WARNING: Missing volumes for hit!" << endl;
178  else
179  cout << "WARNING: Photon hit volume is not the RICH readout plane volume!" << endl;
180  }
181 
182  hit = new PHG4Hitv1();
183 
184  //here we set the entrance values in cm
185  hit->set_x(0, postPoint->GetPosition().x() / cm);
186  hit->set_y(0, postPoint->GetPosition().y() / cm);
187  hit->set_z(0, postPoint->GetPosition().z() / cm);
188 
189  // time in ns
190  hit->set_t(0, postPoint->GetGlobalTime() / nanosecond);
191 
192  //same for exit values (photons absorbed/detected at boundary to post step volume)
193  hit->set_x(1, postPoint->GetPosition().x() / cm);
194  hit->set_y(1, postPoint->GetPosition().y() / cm);
195  hit->set_z(1, postPoint->GetPosition().z() / cm);
196 
197  hit->set_t(1, postPoint->GetGlobalTime() / nanosecond);
198 
199  //set the track ID
200  {
201  hit->set_trkid(aTrack->GetTrackID());
202  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
203  {
204  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
205  {
206  hit->set_trkid(pp->GetUserTrackId());
207  hit->set_shower_id(pp->GetShower()->get_id());
208 
209  pp->SetKeep(true); // we want to keep the track
210  }
211  }
212  }
213 
214  // set optical photon energy deposition
215  hit->set_edep(edep);
216 
217  // Now add the hit
218  hits_->AddHit(sector_id, hit);
219 
220  {
221  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
222  {
223  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
224  {
225  pp->GetShower()->add_g4hit_id(hits_->GetID(), hit->get_hit_id());
226  }
227  }
228  }
229 
230  // return true to indicate the hit was used
231  return true;
232 }
