974 void TpcPrototypeUnpacker::PadPlaneData::Clustering(int zero_suppression, bool verbosity)
975 {
976  using namespace boost;
977  typedef adjacency_list<vecS, vecS, undirectedS> Graph;
978  typedef bimap<Graph::vertex_descriptor, SampleID> VertexList;
979 
980  Graph G;
981  VertexList vertex_list;
982 
983  for (unsigned int pady = 0; pady < kMaxPadY; ++pady)
984  {
985  for (unsigned int padx = 0; padx < kMaxPadX; ++padx)
986  {
987  for (unsigned int sample = 0; sample < kSAMPLE_LENGTH; sample++)
988  {
989  if (m_data[pady][padx][sample] > zero_suppression)
990  {
991  SampleID id{(int) (pady), (int) (padx), (int) (sample)};
992  Graph::vertex_descriptor v = boost::add_vertex(G);
993  vertex_list.insert(VertexList::value_type(v, id));
994 
995  add_edge(v, v, G);
996  }
997  } // for (unsigned int sample = 0; sample < kSAMPLE_LENGTH; sample++)
998  }
999  } // for (unsigned int pady = 0; pady < kMaxPadY; ++pady)
1000 
1001  // connect 2-D adjacent samples within each pad_x
1002  vector<SampleID> search_directions;
1003  search_directions.push_back(SampleID{0, 0, 1});
1004  // search_directions.push_back(SampleID{0, 1, 0});
1005  search_directions.push_back(SampleID{1, 0, 0});
1006 
1007  for (const auto& it : vertex_list.right)
1008  {
1009  const SampleID id = it.first;
1010  const Graph::vertex_descriptor v = it.second;
1011 
1012  for (const SampleID& search_direction : search_directions)
1013  {
1014  // const SampleID next_id = id + search_direction;
1015  SampleID next_id(id);
1016  next_id.adjust(search_direction);
1017 
1018  auto next_it = vertex_list.right.find(next_id);
1019  if (next_it != vertex_list.right.end())
1020  {
1021  add_edge(v, next_it->second, G);
1022  }
1023  }
1024 
1025  } // for (const auto & it : vertex_list)
1026 
1027  // Find the connections between the vertices of the graph (vertices are the rawhits,
1028  // connections are made when they are adjacent to one another)
1029  std::vector<int> component(num_vertices(G));
1030  connected_components(G, &component[0]);
1031 
1032  // Loop over the components(vertices) compiling a list of the unique
1033  // connections (ie clusters).
1034  set<int> comps; // Number of unique components
1035  assert(m_groups.size() == 0); // no overwrite
1036 
1037  for (unsigned int i = 0; i < component.size(); i++)
1038  {
1039  comps.insert(component[i]);
1040  m_groups.insert(make_pair(component[i], vertex_list.left.find(vertex(i, G))->second));
1041  }
1042 
1043  //debug prints
1044  if (verbosity)
1045  for (const int& comp : comps)
1046  {
1047  cout << "TpcPrototypeUnpacker::PadPlaneData::Clustering - find cluster " << comp << " containing ";
1048  const auto range = m_groups.equal_range(comp);
1049 
1050  for (auto iter = range.first; iter != range.second; ++iter)
1051  {
1052  const SampleID& id = iter->second;
1053  cout << "adc[" << id.pady << "][" << id.padx << "][" << id.sample << "] = " << m_data[id.pady][id.padx][id.sample] << ", ";
1054  }
1055  cout << endl;
1056  } // for (const int& comp : comps)
1057 }
