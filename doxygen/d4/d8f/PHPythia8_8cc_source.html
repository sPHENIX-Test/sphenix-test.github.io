185

187  {

189

191  while (!passedGen)
192  {
193  passedGen = _pythia->next();
194  }
195 
196  // test trigger logic
197

199  if (Verbosity() >= VERBOSITY_EVEN_MORE)
200  {
201  cout << "PHPythia8::process_event - triggersize: " << _registeredTriggers.size() << endl;
202  }
203 
204  for (unsigned int tr = 0; tr < _registeredTriggers.size(); tr++)
205  {
206  bool trigResult = _registeredTriggers[tr]->Apply(_pythia);
207 
208  if (Verbosity() >= VERBOSITY_EVEN_MORE)
209  {
210  cout << "PHPythia8::process_event trigger: "
211  << _registeredTriggers[tr]->GetName() << " " << trigResult << endl;
212  }
213 
214  if (_triggersOR && trigResult)
215  {
216  passedTrigger = true;
217  break;
218  }
219  else if (_triggersAND)
220  {
221  andScoreKeeper &= trigResult;
222  }
223 
224  if (Verbosity() >= VERBOSITY_EVEN_MORE && !passedTrigger)
225  {
226  cout << "PHPythia8::process_event - failed trigger: "
227  << _registeredTriggers[tr]->GetName() << endl;
228  }
229  }
230 
231  if ((andScoreKeeper && _triggersAND) || (_registeredTriggers.size() == 0))
232  {
233  passedTrigger = true;
234  genCounter = 0;
235  }
236 
237  passedGen = false;
238  }
239 
240  // fill HepMC object with event & pass to
241 
242  HepMC::GenEvent *genevent = new HepMC::GenEvent(HepMC::Units::GEV, HepMC::Units::MM);
243  _pythiaToHepMC->fill_next_event(*_pythia, genevent, _eventcount);
244 
245  /* pass HepMC to PHNode*/
246  PHHepMCGenEvent *success = hepmc_helper.insert_event(genevent);
247  if (!success)
248  {
249  cout << "PHPythia8::process_event - Failed to add event to HepMC record!" << endl;
250  return Fun4AllReturnCodes::ABORTRUN;
251  }
252 
253  // print outs
254 
255  if (Verbosity() >= VERBOSITY_MORE) cout << "PHPythia8::process_event - FINISHED WHOLE EVENT" << endl;
256  if (_eventcount < 2 && Verbosity() >= VERBOSITY_SOME) _pythia->event.list();
257  if (_eventcount >= 2 && Verbosity() >= VERBOSITY_A_LOT) _pythia->event.list();
258 
259  ++_eventcount;
260 
261  // save statistics
262  if (_integral_node)

264  _integral_node->set_N_Generator_Accepted_Event(_pythia->info.nAccepted());
265  _integral_node->set_N_Processed_Event(_eventcount);
266  _integral_node->set_Sum_Of_Weight(_pythia->info.weightSum());
267  _integral_node->set_Integrated_Lumi(_pythia->info.nAccepted() / (_pythia->info.sigmaGen() * 1e9));
268  }
269 
270  return Fun4AllReturnCodes::EVENT_OK;
271 }
