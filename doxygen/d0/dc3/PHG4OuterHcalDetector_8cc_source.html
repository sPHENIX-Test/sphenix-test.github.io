493 int PHG4OuterHcalDetector::ConstructOuterHcal(G4LogicalVolume *hcalenvelope)
494 {
495  ConsistencyCheck();
496  SetTiltViaNcross(); // if number of crossings is set, use it to determine tilt
497  CheckTiltAngle(); // die if the tilt angle is out of range
498  // the needed steel cutout volume for the magnet is constructed with
499  // the scintillators since we have the theta angle at that point
500 
501  // call field setup here where we have the calculated tilt angle if number
502  // of crossings is given
503  m_FieldSetup = new PHG4OuterHcalFieldSetup(
504  m_NumScintiPlates, /*G4int steelPlates*/
505  m_ScintiGap, /*G4double scintiGap*/
506  m_TiltAngle); /*G4double tiltAngle*/
507 
508  m_ScintiMotherAssembly = ConstructHcalScintillatorAssembly(hcalenvelope);
509 #ifdef SCINTITEST
510  return 0;
511 #endif
512  G4VSolid *steel_plate = ConstructSteelPlate(hcalenvelope);
513  // DisplayVolume(steel_plate_4 ,hcalenvelope);
514  G4LogicalVolume *steel_logical = new G4LogicalVolume(steel_plate, G4Material::GetMaterial(m_Params->get_string_param("material")), "HcalOuterSteelPlate", 0, 0, 0);
515  m_DisplayAction->AddSteelVolume(steel_logical);
516  double phi = 0;
517  double deltaphi = 2 * M_PI / m_NumScintiPlates;
518  ostringstream name;
519  double middlerad = m_OuterRadius - (m_OuterRadius - m_InnerRadius) / 2.;
520  // okay this is crude. Since the inner and outer radius of the scintillator is different from the inner/outer
521  // radius of the steel so the scintillator needs some shifting to get the gaps right
522  // basically the first shift (sumshift) puts it at the inner radius of the steel
523  // then it needs to be shifted up by twice the difference between the inner steel and inner scintillator radius
524  // since sumshift is div by 2 for G4 but we need to shift the full delta(inner radius)
525  double scinti_tile_orig_length = m_ScintiTileXUpper + m_ScintiTileXLower - subtract_from_scinti_x;
526  double shiftup = (m_ScintiInnerRadius - m_InnerRadius) / cos(m_TiltAngle / rad);
527  double sumshift = scinti_tile_orig_length - m_ScintiTileX;
528  sumshift = sumshift - 2 * shiftup;
529  double shiftslat = fabs(m_ScintiTileXLower - m_ScintiTileXUpper) / 2. + sumshift / 2.;
530  // calculate phi offset (copied from code inside following loop):
531  // first get the center point (phi=0) so it's middlerad/0
532  // then shift the scintillator center as documented in loop
533  // then
534  // for positive tilt angles we need the lower left corner of the scintillator
535  // for negative tilt angles we nee the upper right corner of the scintillator
536  // as it turns out the code uses the middle of the face of the scintillator
537  // as reference, if this is a problem the code needs to be modified to
538  // actually calculate the corner (but the math of the construction is that
539  // the middle of the scintillator sits at zero)
540  double xp = cos(phi) * middlerad;
541  double yp = sin(phi) * middlerad;
542  xp -= cos((-m_TiltAngle) / rad - phi) * shiftslat;
543  yp += sin((-m_TiltAngle) / rad - phi) * shiftslat;
544  if (m_TiltAngle > 0)
545  {
546  double xo = xp - (m_ScintiTileX / 2.) * cos(m_TiltAngle / rad);
547  double yo = yp - (m_ScintiTileX / 2.) * sin(m_TiltAngle / rad);
548  phi = -atan(yo / xo);
549  }
550  else if (m_TiltAngle < 0)
551  {
552  double xo = xp + (m_ScintiTileX / 2.) * cos(m_TiltAngle / rad);
553  double yo = yp + (m_ScintiTileX / 2.) * sin(m_TiltAngle / rad);
554  phi = -atan(yo / xo);
555  }
556  // else (for m_TiltAngle = 0) phi stays zero
557  for (int i = 0; i < m_NumScintiPlates; i++)
558  {
559  G4RotationMatrix *Rot = new G4RotationMatrix();
560  double ypos = sin(phi) * middlerad;
561  double xpos = cos(phi) * middlerad;
562  // the center of the scintillator is not the center of the inner hcal
563  // but depends on the tilt angle. Therefore we need to shift
564  // the center from the mid point
565  ypos += sin((-m_TiltAngle) / rad - phi) * shiftslat;
566  xpos -= cos((-m_TiltAngle) / rad - phi) * shiftslat;
567  Rot->rotateZ(phi * rad + m_TiltAngle);
568  G4ThreeVector g4vec(xpos, ypos, 0);
569  // great the MakeImprint always adds 1 to the copy number and 0 has a
570  // special meaning (which then also adds 1). Basically our volume names
571  // will start at 1 instead of 0 and there is nothing short of patching this
572  // method. I'll take care of this in the decoding of the volume name
573  // AAAAAAARHGS
574  m_ScintiMotherAssembly->MakeImprint(hcalenvelope, g4vec, Rot, i, OverlapCheck());
575  delete Rot;
576  Rot = new G4RotationMatrix();
577  Rot->rotateZ(-phi * rad);
578  name.str("");
579  name << "OuterHcalSteel_" << i;
580  m_SteelAbsorberVec.insert(new G4PVPlacement(Rot, G4ThreeVector(0, 0, 0), steel_logical, name.str().c_str(), hcalenvelope, 0, i, OverlapCheck()));
581  phi += deltaphi;
582  }
583  hcalenvelope->SetFieldManager(m_FieldSetup->get_Field_Manager_Gap(), false);
584 
585  steel_logical->SetFieldManager(m_FieldSetup->get_Field_Manager_Iron(), true);
586  return 0;

872  PHG4OuterHcalDetector::Point_2 pin1(m_InnerRadius, 0), pin2(0, m_InnerRadius), pin3(-m_InnerRadius, 0);
873  Circle_2 inner_circle(pin1, pin2, pin3);
874  PHG4OuterHcalDetector::Point_2 pmid1(mid_radius, 0), pmid2(0, mid_radius), pmid3(-mid_radius, 0);
875  Circle_2 mid_circle(pmid1, pmid2, pmid3);
876  PHG4OuterHcalDetector::Point_2 pout1(m_OuterRadius, 0), pout2(0, m_OuterRadius), pout3(-m_OuterRadius, 0);
877  Circle_2 outer_circle(pout1, pout2, pout3);
878  Line_2 l_up(pnull, phightmp);
879  vector<CGAL::Object> res;
880  CGAL::intersection(outer_circle, l_up, std::back_inserter(res));
881  PHG4OuterHcalDetector::Point_2 upperright;
882  vector<CGAL::Object>::const_iterator iter;
883  for (iter = res.begin(); iter != res.end(); ++iter)
884  {
885  CGAL::Object obj = *iter;
886  if (const std::pair<CGAL::Circular_arc_point_2<PHG4OuterHcalDetector::Circular_k>, unsigned> *point = CGAL::object_cast<std::pair<CGAL::Circular_arc_point_2<PHG4OuterHcalDetector::Circular_k>, unsigned>>(&obj))
887  {
888  if (CGAL::to_double(point->first.x()) > 0)
889  {
890  PHG4OuterHcalDetector::Point_2 pntmp(CGAL::to_double(point->first.x()), CGAL::to_double(point->first.y()));
891  upperright = pntmp;
892  }
893  }
894  else
895  {
896  cout << "CGAL::Object type not pair..." << endl;
897  exit(1);
898  }
899  }
900  Line_2 l_right(plow, upperright);
901  res.clear();
902  PHG4OuterHcalDetector::Point_2 midpoint;
903  CGAL::intersection(mid_circle, l_right, std::back_inserter(res));
904  for (iter = res.begin(); iter != res.end(); ++iter)
905  {
906  CGAL::Object obj = *iter;
907  if (const std::pair<CGAL::Circular_arc_point_2<PHG4OuterHcalDetector::Circular_k>, unsigned> *point = CGAL::object_cast<std::pair<CGAL::Circular_arc_point_2<PHG4OuterHcalDetector::Circular_k>, unsigned>>(&obj))
908  {
909  if (CGAL::to_double(point->first.x()) > 0)
910  {
911  PHG4OuterHcalDetector::Point_2 pntmp(CGAL::to_double(point->first.x()), CGAL::to_double(point->first.y()));
912  midpoint = pntmp;
913  }
914  }
915  else
916  {
917  cout << "CGAL::Object type not pair..." << endl;
918  exit(1);
919  }
920  }
921  // length left side
922  double ll = sqrt((CGAL::to_double(midpoint.x()) - m_InnerRadius) * (CGAL::to_double(midpoint.x()) - m_InnerRadius) + CGAL::to_double(midpoint.y()) * CGAL::to_double(midpoint.y()));
923  double upside = sqrt(CGAL::to_double(midpoint.x()) * CGAL::to_double(midpoint.x()) + CGAL::to_double(midpoint.y()) * CGAL::to_double(midpoint.y()));
924  // c^2 = a^2+b^2 - 2ab*cos(gamma)
925  // gamma = acos((a^2+b^2=c^2)/2ab
926  double tiltangle = acos((ll * ll + upside * upside - m_InnerRadius * m_InnerRadius) / (2 * ll * upside));
927  tiltangle = tiltangle * rad;
928  m_TiltAngle = copysign(tiltangle, ncross);
929  m_Params->set_double_param("tilt_angle", m_TiltAngle / deg);
930  return;
931 }

933 // check if tilt angle is reasonable - too large, no intersections with inner radius
934 int PHG4OuterHcalDetector::CheckTiltAngle() const
935 {
936  if (fabs(m_TiltAngle) >= M_PI)
937  {
938  cout << PHWHERE << "invalid tilt angle, abs(tilt) >= 90 deg: " << (m_TiltAngle / deg)
939  << endl;
940  exit(1);
941  }
942 
943  double mid_radius = m_InnerRadius + (m_OuterRadius - m_InnerRadius) / 2.;
944  PHG4OuterHcalDetector::Point_2 pmid(mid_radius, 0); // center of scintillator
945  double xcoord = 0;
946  double ycoord = mid_radius * tan(m_TiltAngle / rad);
947  PHG4OuterHcalDetector::Point_2 pxnull(xcoord, ycoord);
948  Line_2 s2(pmid, pxnull);
949  PHG4OuterHcalDetector::Point_2 sc1(m_InnerRadius, 0), sc2(0, m_InnerRadius), sc3(-m_InnerRadius, 0);
950  Circle_2 inner_circle(sc1, sc2, sc3);
951  vector<CGAL::Object> res;
952  CGAL::intersection(inner_circle, s2, std::back_inserter(res));
953  if (res.size() == 0)
954  {
955  cout << PHWHERE << " Tilt angle " << (m_TiltAngle / deg)
956  << " too large, no intersection with inner radius" << endl;
957  exit(1);
958  }
959  return 0;
960 }
