108 bool PHG4OuterHcalSteppingAction::UserSteppingAction(const G4Step* aStep, bool)
109 {
110  G4TouchableHandle touch = aStep->GetPreStepPoint()->GetTouchableHandle();
111  G4TouchableHandle touchpost = aStep->GetPostStepPoint()->GetTouchableHandle();
112  // get volume of the current step
113  G4VPhysicalVolume* volume = touch->GetVolume();
114 
115  // m_Detector->IsInOuterHcal(volume)
116  // returns
117  // 0 is outside of OuterHcal
118  // 1 is inside scintillator
119  // -1 is steel absorber (if absorber set to active)
120 
121  int whichactive = m_Detector->IsInOuterHcal(volume);
122 
123  if (!whichactive)
124  {
125  return false;
126  }
127 
128  if (m_EnableFieldCheckerFlag)
129  {
130  FieldChecker(aStep);
131  }
132 
133  int layer_id = -1;
134  int tower_id = -1;
135  if (whichactive > 0) // scintillator
136  {
137  pair<int, int> layer_tower = m_Detector->GetLayerTowerId(volume);
138  layer_id = layer_tower.first;
139  tower_id = layer_tower.second;
140  }
141  else
142  {
143  layer_id = touch->GetCopyNumber(); // steel plate id
144  }
145  // collect energy and track length step by step
146  G4double edep = aStep->GetTotalEnergyDeposit() / GeV;
147  G4double eion = (aStep->GetTotalEnergyDeposit() - aStep->GetNonIonizingEnergyDeposit()) / GeV;
148  G4double light_yield = 0;
149 
150  const G4Track* aTrack = aStep->GetTrack();
151 
152  // if this block stops everything, just put all kinetic energy into edep
153  if (m_IsBlackHoleFlag)
154  {
155  edep = aTrack->GetKineticEnergy() / GeV;
156  G4Track* killtrack = const_cast<G4Track*>(aTrack);
157  killtrack->SetTrackStatus(fStopAndKill);
158  }
159 
160  // make sure we are in a volume
161  if (m_IsActiveFlag)
162  {
163  bool geantino = false;
164 
165  // the check for the pdg code speeds things up, I do not want to make
166  // an expensive string compare for every track when we know
167  // geantino or chargedgeantino has pid=0
168  if (aTrack->GetParticleDefinition()->GetPDGEncoding() == 0 &&
169  aTrack->GetParticleDefinition()->GetParticleName().find("geantino") != string::npos)
170  {
171  geantino = true;
172  }
173  G4StepPoint* prePoint = aStep->GetPreStepPoint();
174  G4StepPoint* postPoint = aStep->GetPostStepPoint();
175  // cout << "track id " << aTrack->GetTrackID() << endl;
176  // cout << "time prepoint: " << prePoint->GetGlobalTime() << endl;
177  // cout << "time postpoint: " << postPoint->GetGlobalTime() << endl;
178  switch (prePoint->GetStepStatus())
179  {
180  case fPostStepDoItProc:
181  if (m_SavePostStepStatus != fGeomBoundary)
182  {
183  break;
184  }
185  else
186  {
187  cout << GetName() << ": New Hit for " << endl;
188  cout << "prestep status: " << PHG4StepStatusDecode::GetStepStatus(prePoint->GetStepStatus())
189  << ", poststep status: " << PHG4StepStatusDecode::GetStepStatus(postPoint->GetStepStatus())
190  << ", last pre step status: " << PHG4StepStatusDecode::GetStepStatus(m_SavePreStepStatus)
191  << ", last post step status: " << PHG4StepStatusDecode::GetStepStatus(m_SavePostStepStatus) << endl;
192  cout << "last track: " << m_SaveTrackId
193  << ", current trackid: " << aTrack->GetTrackID() << endl;
194  cout << "phys pre vol: " << volume->GetName()
195  << " post vol : " << touchpost->GetVolume()->GetName() << endl;
196  cout << " previous phys pre vol: " << m_SaveVolPre->GetName()
197  << " previous phys post vol: " << m_SaveVolPost->GetName() << endl;
198  }
199  case fGeomBoundary:
200  case fUndefined:
201  if (!m_Hit)
202  {
203  m_Hit = new PHG4Hitv1();
204  }
205  //here we set the entrance values in cm
206  m_Hit->set_x(0, prePoint->GetPosition().x() / cm);
207  m_Hit->set_y(0, prePoint->GetPosition().y() / cm);
208  m_Hit->set_z(0, prePoint->GetPosition().z() / cm);
209  // time in ns
210  m_Hit->set_t(0, prePoint->GetGlobalTime() / nanosecond);
211  //set the track ID
212  m_Hit->set_trkid(aTrack->GetTrackID());
213  m_SaveTrackId = aTrack->GetTrackID();
214  //set the initial energy deposit
215  m_Hit->set_edep(0);
216  if (whichactive > 0) // return of IsInOuterHcalDetector, > 0 hit in scintillator, < 0 hit in absorber
217  {
218  m_Hit->set_scint_id(tower_id); // the slat id
219  m_Hit->set_eion(0);
220  m_Hit->set_light_yield(0); // for scintillator only, initialize light yields
221  // Now save the container we want to add this hit to
222  m_SaveHitContainer = m_Hits;
223  }
224  else
225  {
226  m_SaveHitContainer = m_AbsorberHits;
227  }
228  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
229  {
230  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
231  {
232  m_Hit->set_trkid(pp->GetUserTrackId());
233  m_Hit->set_shower_id(pp->GetShower()->get_id());
234  m_SaveShower = pp->GetShower();
235  }
236  }
237  break;
238  default:
239  break;
240  }
241  // some sanity checks for inconsistencies
242  // check if this hit was created, if not print out last post step status
243  if (!m_Hit || !isfinite(m_Hit->get_x(0)))
244  {
245  cout << GetName() << ": hit was not created" << endl;
246  cout << "prestep status: " << PHG4StepStatusDecode::GetStepStatus(prePoint->GetStepStatus())
247  << ", poststep status: " << PHG4StepStatusDecode::GetStepStatus(postPoint->GetStepStatus())
248  << ", last pre step status: " << PHG4StepStatusDecode::GetStepStatus(m_SavePreStepStatus)
249  << ", last post step status: " << PHG4StepStatusDecode::GetStepStatus(m_SavePostStepStatus) << endl;
250  cout << "last track: " << m_SaveTrackId
251  << ", current trackid: " << aTrack->GetTrackID() << endl;
252  cout << "phys pre vol: " << volume->GetName()
253  << " post vol : " << touchpost->GetVolume()->GetName() << endl;
254  cout << " previous phys pre vol: " << m_SaveVolPre->GetName()
255  << " previous phys post vol: " << m_SaveVolPost->GetName() << endl;
256  gSystem->Exit(1);
257  }
258  m_SavePostStepStatus = postPoint->GetStepStatus();
259  // check if track id matches the initial one when the hit was created
260  if (aTrack->GetTrackID() != m_SaveTrackId)
261  {
262  cout << GetName() << ": hits do not belong to the same track" << endl;
263  cout << "saved track: " << m_SaveTrackId
264  << ", current trackid: " << aTrack->GetTrackID()
265  << endl;
266  gSystem->Exit(1);
267  }
268  m_SavePreStepStatus = prePoint->GetStepStatus();
269  m_SavePostStepStatus = postPoint->GetStepStatus();
270  m_SaveVolPre = volume;
271  m_SaveVolPost = touchpost->GetVolume();
272  // here we just update the exit values, it will be overwritten
273  // for every step until we leave the volume or the particle
274  // ceases to exist
275  m_Hit->set_x(1, postPoint->GetPosition().x() / cm);
276  m_Hit->set_y(1, postPoint->GetPosition().y() / cm);
277  m_Hit->set_z(1, postPoint->GetPosition().z() / cm);
278 
279  m_Hit->set_t(1, postPoint->GetGlobalTime() / nanosecond);
280 
281  if (whichactive > 0)
282  {
283  if (m_LightScintModelFlag)
284  {
285  light_yield = GetVisibleEnergyDeposition(aStep);
286  }
287  else
288  {
289  light_yield = eion;
290  }
291 
292  if (ValidCorrection())
293  {
294 double cor = GetLightCorrection(postPoint->GetPosition().x() , (postPoint->GetPosition().y() ));
295 cout << "applying cor: " << cor << endl;
296  light_yield = light_yield * GetLightCorrection(postPoint->GetPosition().x() , (postPoint->GetPosition().y() ));
297  }
298  }
299 
300  //sum up the energy to get total deposited
301  m_Hit->set_edep(m_Hit->get_edep() + edep);
302  if (whichactive > 0)
303  {
304  m_Hit->set_eion(m_Hit->get_eion() + eion);
305  m_Hit->set_light_yield(m_Hit->get_light_yield() + light_yield);
306  }
307  if (geantino)
308  {
309  m_Hit->set_edep(-1); // only energy=0 g4hits get dropped, this way geantinos survive the g4hit compression
310  m_Hit->set_eion(-1);
311  }
312  if (edep > 0)
313  {
314  if (G4VUserTrackInformation* p = aTrack->GetUserInformation())
315  {
316  if (PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p))
317  {
318  pp->SetKeep(1); // we want to keep the track
319  }
320  }
321  }
322 
323  // if any of these conditions is true this is the last step in
324  // this volume and we need to save the hit
325  // postPoint->GetStepStatus() == fGeomBoundary: track leaves this volume
326  // postPoint->GetStepStatus() == fWorldBoundary: track leaves this world
327  // (happens when your detector goes outside world volume)
328  // postPoint->GetStepStatus() == fAtRestDoItProc: track stops (typically
329  // aTrack->GetTrackStatus() == fStopAndKill is also set)
330  // aTrack->GetTrackStatus() == fStopAndKill: track ends
331  if (postPoint->GetStepStatus() == fGeomBoundary ||
332  postPoint->GetStepStatus() == fWorldBoundary ||
333  postPoint->GetStepStatus() == fAtRestDoItProc ||
334  aTrack->GetTrackStatus() == fStopAndKill)
335  {
336  // save only hits with energy deposit (or -1 for geantino)
337  if (m_Hit->get_edep())
338  {
339  m_SaveHitContainer->AddHit(layer_id, m_Hit);
340  if (m_SaveShower)
341  {
342  m_SaveShower->add_g4hit_id(m_SaveHitContainer->GetID(), m_Hit->get_hit_id());
343  }
344  // ownership has been transferred to container, set to null
345  // so we will create a new hit for the next track
346  m_Hit = nullptr;
347  }
348  else
349  {
350  // if this hit has no energy deposit, just reset it for reuse
351  // this means we have to delete it in the dtor. If this was
352  // the last hit we processed the memory is still allocated
353  m_Hit->Reset();
354  }
355  }
356  // return true to indicate the hit was used
357  return true;
358  }
359  else
360  {
361  return false;
362  }
