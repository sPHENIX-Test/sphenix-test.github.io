150 int PHRaveVertexing::process_event(PHCompositeNode* topNode)
151 {
152  _event++;
153 
154  if (Verbosity() > 1)
155  std::cout << PHWHERE << "Events processed: " << _event << std::endl;
156 
157  GetNodes(topNode);
158 
160  GenFitTrackMap gf_track_map;
161  vector<genfit::Track*> gf_tracks;
162  if (Verbosity() > 1) _t_translate->restart();
163  for (SvtxTrackMap::Iter iter = _trackmap->begin(); iter != _trackmap->end();
164  ++iter)
165  {
166  SvtxTrack* svtx_track = iter->second;
167  if (!svtx_track)
168  continue;
169 
170  if (!(svtx_track->get_ndf() >= _vertex_min_ndf))
171  continue;
172 
173  //auto genfit_track = shared_ptr<genfit::Track> (TranslateSvtxToGenFitTrack(svtx_track));
174  auto genfit_track = TranslateSvtxToGenFitTrack(svtx_track);
175  if (!genfit_track)
176  continue;
177  gf_track_map.insert({genfit_track, iter->first});
178  gf_tracks.push_back(const_cast<genfit::Track*>(genfit_track));
179  }
180  if (Verbosity() > 1) _t_translate->stop();
181 
182  if (Verbosity() > 1) _t_rave->restart();
183  vector<genfit::GFRaveVertex*> rave_vertices;
184  if (gf_tracks.size() >= 2)
185  {
186  try
187  {
188  _vertex_finder->findVertices(&rave_vertices, gf_tracks);
189  }
190  catch (...)
191  {
192  if (Verbosity() > 1)
193  std::cout << PHWHERE << "GFRaveVertexFactory::findVertices failed!";
194  }
195  }
196  if (Verbosity() > 1) _t_rave->stop();
197  FillSvtxVertexMap(rave_vertices, gf_track_map);
198 
199  for (auto iter : gf_track_map) delete iter.first;
200 
201  if (Verbosity() > 1)
202  {
203  _vertexmap->identify();
204 
205  std::cout << "=============== Timers: ===============" << std::endl;
206  std::cout << "Event: " << _event << std::endl;
207  std::cout << "Translate: " << _t_translate->get_accumulated_time() / 1000. << " sec" << std::endl;
208  std::cout << "RAVE: " << _t_rave->get_accumulated_time() / 1000. << " sec" << std::endl;
209  std::cout << "=======================================" << std::endl;
210  }
211 
212  return Fun4AllReturnCodes::EVENT_OK;
213 }

412 genfit::Track* PHRaveVertexing::TranslateSvtxToGenFitTrack(SvtxTrack* svtx_track)
413 {
414  try
415  {
416  // The first state is extracted to PCA, second one is the one with measurement
417  SvtxTrackState* svtx_state(nullptr);
418  //SvtxTrackState* svtx_state = (svtx_track->begin_states())->second;
419 
420  if (svtx_track->begin_states() == svtx_track->end_states())
421  {
422  LogDebug("TranslateSvtxToGenFitTrack no state in track!");
423  return nullptr;
424  }
425  else if (++(svtx_track->begin_states()) == svtx_track->end_states())
426  {
427  // only one state in track
428  svtx_state = (svtx_track->begin_states())->second;
429  }
430  else
431  {
432  // multiple state in track
433  // The first state is extracted to PCA, second one is the one with measurement
434  svtx_state = (++(svtx_track->begin_states()))->second;
435  }
436 
437  if (!svtx_state)
438  {
439  LogDebug("TranslateSvtxToGenFitTrack invalid state found on track!");
440  return nullptr;
441  }
442 
443  TVector3 pos(svtx_state->get_x(), svtx_state->get_y(), svtx_state->get_z());
444  TVector3 mom(svtx_state->get_px(), svtx_state->get_py(), svtx_state->get_pz());
445  TMatrixDSym cov(6);
446  for (int i = 0; i < 6; ++i)
447  {
448  for (int j = 0; j < 6; ++j)
449  {
450  cov[i][j] = svtx_state->get_error(i, j);
451  }
452  }
453 
454 #ifdef _DEBUG_
455  {
456  cout << "DEBUG" << __LINE__ << endl;
457  cout << "path length: " << svtx_state->get_pathlength() << endl;
458  cout << "radius: " << pos.Perp() << endl;
459  cout << "DEBUG: " << __LINE__ << endl;
460  for (int i = 0; i < 6; ++i)
461  {
462  for (int j = 0; j < 6; ++j)
463  {
464  cout << svtx_state->get_error(i, j) << "\t";
465  }
466  cout << endl;
467  }
468 
469  cov.Print();
470  }
471 
472 #endif
473 
474  genfit::AbsTrackRep* rep = new genfit::RKTrackRep(_primary_pid_guess);
475  genfit::Track* genfit_track = new genfit::Track(rep, TVector3(0, 0, 0), TVector3(0, 0, 0));
476 
477  genfit::FitStatus* fs = new genfit::FitStatus();
478  fs->setCharge(svtx_track->get_charge());
479  fs->setChi2(svtx_track->get_chisq());
480  fs->setNdf(svtx_track->get_ndf());
481  fs->setIsFitted(true);
482  fs->setIsFitConvergedFully(true);
483 
484  genfit_track->setFitStatus(fs, rep);

490 
491  genfit::MeasuredStateOnPlane* ms = new genfit::MeasuredStateOnPlane(rep);

493 #ifdef _DEBUG_

495  cout << "DEBUG: " << __LINE__ << endl;
496  ms->Print();
497  cout << "Orig: " << __LINE__ << endl;
498  cov.Print();
499  cout << "Translate: " << __LINE__ << endl;

504 
505  //< Acording to the special order of using the stored states

508  genfit_track->insertPoint(tp);
