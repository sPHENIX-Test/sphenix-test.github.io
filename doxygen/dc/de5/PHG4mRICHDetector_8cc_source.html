195 void PHG4mRICHDetector::LensPar::Set_halfXYZ(G4double halfX, G4double grooveDensity)
196 {
197  halfXYZ[0] = halfX;
198  halfXYZ[1] = halfXYZ[0];
199 
200  G4double NumberOfGrooves = floor(grooveDensity * (eff_diameter / 2.0));
201  G4double Rmin1 = (NumberOfGrooves - 1) * (grooveWidth);
202  G4double Rmax1 = (NumberOfGrooves - 0) * (grooveWidth);
203  halfXYZ[2] = (GetSagita(Rmax1) - GetSagita(Rmin1) + centerThickness) / 2.0;
204 }

206 G4double PHG4mRICHDetector::LensPar::GetSagita(G4double r)
207 {
208  G4double Conic = -1.0; // original:
209  //G4int lens_type = 3;
210  G4int lens_type = 5;
211  G4double Curvature;
212  G4double Aspher[8] = {0, 0, 0, 0, 0, 0, 0, 0};
213 
214  if (lens_type == 1)
215  {
216  Curvature = 0.00437636761488 / mm;
217  Aspher[0] = 4.206739256e-05 / (mm);
218  Aspher[1] = 9.6440152e-10 / (mm3);
219  Aspher[2] = -1.4884317e-15 / (mm2 * mm3);
220  }
221  else if (lens_type == 2)
222  {
223  Curvature = 0.0132 / mm; // r=77mm, f~14cm
224  Aspher[0] = 32.0e-05 / (mm);
225  Aspher[1] = -2.0e-7 / (mm3);
226  Aspher[2] = 1.2e-13 / (mm2 * mm3);
227  }
228  else if (lens_type == 3)
229  {
230  Curvature = 0.0150 / mm; // r=77mm, f~12.5cm
231  Aspher[0] = 42.0e-05 / (mm);
232  Aspher[1] = -3.0e-7 / (mm3);
233  Aspher[2] = 1.2e-13 / (mm2 * mm3);
234  }
235  else if (lens_type == 4)
236  {
237  Curvature = 0.0175 / mm; // r=77mm, f~10cm
238  Aspher[0] = 72.0e-05 / (mm);
239  Aspher[1] = -5.0e-7 / (mm3);
240  Aspher[2] = 1.2e-13 / (mm2 * mm3);
241  }
242  else if (lens_type == 5)
243  {
244  Curvature = 1 / (f * (n - 1));
245  }
246 
247  G4double TotAspher = 0.0 * mm;
248  for (G4int k = 1; k < 9; k++)
249  {
250  TotAspher += Aspher[k - 1] * std::pow(r, 2 * k);
251  }
252 
253  G4double ArgSqrt = 1.0 - (1.0 + Conic) * std::pow(Curvature, 2) * std::pow(r, 2); // note conic=-1, so ArgSqrt = 1.0
254 
255  if (ArgSqrt < 0.0)
256  {
257  G4cout << "UltraFresnelLensParameterisation::Sagita: Square Root of <0 !" << G4endl;
258  }
259 
260  G4double Sagita_value = Curvature * std::pow(r, 2) / (1.0 + std::sqrt(ArgSqrt)) + TotAspher;
261  return Sagita_value;
262 }

738 void PHG4mRICHDetector::build_lens(LensPar* par, G4LogicalVolume* motherLV)
739 {
740  const G4int NumberOfGrooves = floor((par->eff_diameter / 2.0) / par->grooveWidth);
741  G4Polycone* Groove_poly[NumberOfGrooves];
742  G4LogicalVolume* Groove_log[NumberOfGrooves];
743 
744  G4VisAttributes* SurfaceVisAtt = new G4VisAttributes(G4Colour(0.0, 1.0, 1.0));
745  SurfaceVisAtt->SetVisibility(true);
746  SurfaceVisAtt->SetForceWireframe(true);
747  SurfaceVisAtt->SetForceSolid(true);
748 
749  int igroove;
750  for (igroove = 0; igroove < 1000; igroove++)
751  { //just put a arbitrary large number
752  //--------------------------------
753  //Grooves' inner and outer radius
754  //--------------------------------
755  G4double iRmin1 = (igroove + 0) * par->grooveWidth;
756  G4double iRmax1 = (igroove + 1) * par->grooveWidth;
757  G4double iRmin2 = iRmin1;
758  G4double iRmax2 = iRmin2 + 0.0001;
759 
760  G4double lens_poly_rmin[3] = {iRmin1, iRmin1, iRmin2};
761  G4double lens_poly_rmax[3] = {iRmax1, iRmax1, iRmax2};
762 
763  if (iRmax1 > par->diameter / 2.0) break; //if iRmax1>Lens radius (outside the lens), break
764 
765  //--------------------------------
766  //phi angle
767  //--------------------------------
768  G4double phi1;
769  G4double phi2;
770  G4double deltaPhi;
771 
772  if (iRmax1 < par->halfXYZ[0])
773  { //draw a full circle
774  phi1 = 0; //in rad
775  deltaPhi = twopi; //in rad. two pi
776  }
777  else
778  {
779  phi1 = acos(par->halfXYZ[0] / iRmax1); //in rad
780  phi2 = asin(par->halfXYZ[0] / iRmax1); //in rad, assume lens is square -> halfy=halfx
781  deltaPhi = phi2 - phi1;
782  }
783  //--------------------------------
784  //grooves profile
785  //--------------------------------
786  G4double lens_poly_z[3];
787  int numOfLayer;
788 
789  if (iRmin1 < par->eff_diameter / 2.0)
790  { //if iRmin>=effective radius, dZ=0, i.e. flat
791  numOfLayer = 3;
792  G4double dZ = par->GetSagita(iRmax1) - par->GetSagita(iRmin1);
793  lens_poly_z[0] = par->halfXYZ[2];
794  lens_poly_z[1] = -par->halfXYZ[2] + dZ;
795  lens_poly_z[2] = -par->halfXYZ[2];
796  }
797  else
798  {
799  numOfLayer = 2;
800  lens_poly_z[0] = par->halfXYZ[2];
801  lens_poly_z[1] = par->halfXYZ[2] - par->centerThickness;
802  lens_poly_z[2] = 0;
803  }
804 
805  //--------------------------------
806  //build grooves
807  //--------------------------------
808  int repeat = 1;
809  if (iRmax1 >= par->halfXYZ[0])
810  {
811  repeat = 4;
812  } //4 edges
813  for (int i = 0; i < repeat; i++)
814  {
815  Groove_poly[i] = new G4Polycone(par->name.c_str(), phi1, deltaPhi, numOfLayer, lens_poly_z, lens_poly_rmin, lens_poly_rmax);
816  Groove_log[i] = new G4LogicalVolume(Groove_poly[i], par->material, par->name.c_str(), 0, 0, 0);
817  new G4PVPlacement(0, par->pos, Groove_log[i], par->name.c_str(), motherLV, false, 0, OverlapCheck());
818 
819  Groove_log[i]->SetVisAttributes(SurfaceVisAtt);
820  phi1 = phi1 + halfpi; //g4 pre-defined: halfpi=pi/2
821  }
822  }
823 }
